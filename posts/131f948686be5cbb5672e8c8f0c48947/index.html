getLocationID(city,area){
  var that=this
  wx.request({
    url: 'https://geoapi.qweather.com/v2/city/lookup?',
    method:"GET",
    data:{
    location:area,
    adm:city,
    key:'00ecc66f369c457896b11aae1a4f94d3'
    },
