Rectype R[n];  
{  
    int i = 1;  
    do {  
        exchang = False;  
        for (j = n; j >= 2; j--) {  
            if (R[j] < R[j - 1]) {  
                temp = R[j - 1];  
                R[j - 1] = R[j];  
                R[j] = temp;  
                exchang = true;  
            }  
          break;
        }  
    } while (exchang==false);  
}
