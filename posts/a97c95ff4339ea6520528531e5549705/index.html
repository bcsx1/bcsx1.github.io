import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    shoplistitems: [],
    sum: 0,
  },
  getters: {
    getSum(state) {
      state.sum = 0;
      for (var i = 0; i < state.shoplistitems.length; i++) {
        if (state.shoplistitems[i].visible == true) {
          state.sum += state.shoplistitems[i].price * state.shoplistitems[i].customerNum
        }
      }
      return state.sum
    },
  },
})

Mutation

同步操作

import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    str: '今天学vueX，这是store传过来的数据',
    num: 0,
    shoplistitems: [],
    sum: 0,
    all: true
  },
  mutations: {
    mutationAddnum(state, n = 0) {
      state.num += n
    },
    mutationRednum(state, n = 0) {
      state.num -= n
    },
    mutationAddShopitem(state, shopitem) {
      for (var i = 0; i < state.shoplistitems.length; i++) {
        if (state.shoplistitems[i].goodsName == shopitem.goodsName) {
          state.shoplistitems[i].customerNum++
          return
        }
        
      }
      // if (state.shoplistitems.indexOf(shopitem) == -1) {
        state.shoplistitems = [shopitem, ...state.shoplistitems];
      // }
    },
    mutationDelShopitem(state, index) {
      state.shoplistitems.splice(index, 1)
    },
    shopcartNumAdd(state, index) {
      state.shoplistitems[index].customerNum++;
    },
    shopcartNumdel(state, index) {
      state.shoplistitems[index].customerNum--;
      if (state.shoplistitems[index].customerNum == 0) {
        state.shoplistitems.splice(index, 1)
      }
    },
    checkShoplist(state, index) {
      state.shoplistitems[index].visible = !state.shoplistitems[index].visible
    },
    allcheck(state) {
      state.all = !state.all
      if (state.all == true) {
        for (var i = 0; i < state.shoplistitems.length; i++) {
          state.shoplistitems[i].visible = true
        }
      } else {
        for (var i = 0; i < state.shoplistitems.length; i++) {
          state.shoplistitems[i].visible = false
        }
      }
    }
  },
 })

Action

针对mutations的异步操作，使用commit关键

import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    str: '今天学vueX，这是store传过来的数据',
    num: 0,
    shoplistitems: [],
    sum: 0,
    all: true
  },
  mutations: {
    mutationAddnum(state, n = 0) {
      state.num += n
    },
    mutationAddShopitem(state, shopitem) {
      for (var i = 0; i < state.shoplistitems.length; i++) {
        if (state.shoplistitems[i].goodsName == shopitem.goodsName) {
          state.shoplistitems[i].customerNum++
          return
        }
      }
      // if (state.shoplistitems.indexOf(shopitem) == -1) {
        state.shoplistitems = [shopitem, ...state.shoplistitems];
      // }
    },
    mutationDelShopitem(state, index) {
      state.shoplistitems.splice(index, 1)
    },
  },
  actions: {
    actionAddnum({ commit }, n = 0) {
      commit('mutationAddnum', n)
    },
    actionAddShopitem(context, shopitem) {
      context.commit('mutationAddShopitem', shopitem)
    },
    actionDelShoplist({ commit }, index) {
      commit('mutationDelShopitem', index)
    }
  },
 })

模块化store

import Vue from 'vue'
import Vuex from 'vuex'

import user from './module/user'
import app from './module/app'

const context = require.context('./mod', false, /\.ts$/);
const moduleStores = {};
context.keys().forEach(key => {
  const fileName = key.slice(2, -3);
  const fileModule = context(key).default;
  moduleStores[fileName] = {
    ...fileModule,
    namespaced: true,
  }
})
export default new Vuex.Store({
  modules: {
    user,
    app,
    ...moduleStores,
  },
  getters
})
