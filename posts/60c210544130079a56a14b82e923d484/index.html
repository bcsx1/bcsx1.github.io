3.运算符

        //1.算术运算符：+:正数 加法运算 连接
        int a=5;
        int b=6;
        System.out.println(+a);//+5
        System.out.println(a+b);//11
        System.out.println("a:"+a);//a:5
        System.out.println("abc"+"def");//abcdef
        //-:负数 减法运算
        System.out.println(-a);//-5
        System.out.println(b-a);//1
        // *:乘法运算
        System.out.println(b*a);//300
        // /:除法运算(取商) %：除法运算（取余）
        System.out.println(b/a);//1
        System.out.println(b%a);//1
        //++
        System.out.println(++b);//b=b+1 b++ 7
        //--
        System.out.println(--b);//b=b-1  b-- 6
        int x=5;
        int y=x++ + 6;//y=11 x=6
        int z=++x + 5;//x=7 z=12
        System.out.println("y="+y);
        System.out.println("z="+z);
        //2.比较运算符：== != > < >= <=
        //== 数值==数值 布尔==布尔 引用类型==引用类型
        System.out.println(1==1);
        System.out.println(true==false);
        System.out.println("hui"=="hui");
        //!= 数值!=数值 布尔!=布尔 引用类型!=引用类型
        //> < >= <=只能用于数值与数值进行比较 可以包含字符
        //3.赋值运算符:= 赋值 注意左右两边的数据类型是否一致，不一致就报错
        int c=6;
        //错误：int c1=c+5.5;
        double c1=c+5.5;
        System.out.println("c1="+c1);
        //扩展赋值运算符：+= -= *= /= %=,以+=为例
        short d=5;
        //d=d+1;这样写会出错
        d+=1;//这样写不会出错,short=short(short+int) +=符号可以进行隐式数据类型转换
        System.out.println("d="+d);
        //4.逻辑运算符：& && | || ! ^
        //& &&全true则true，否则false
        int e=6;
        int f=7;
        System.out.println(e>7&f++>7);
        System.out.println("f="+f);//f=8
        System.out.println(e>7&&f++>7);
        System.out.println("f="+f);//f=8
        //| ||全false则false，否则true
        int e1=6;
        int f1=7;
        System.out.println(e1<7|f1++>7);
        System.out.println("f1="+f1);//f1=8
        System.out.println(e1<7||f1++>7);
        System.out.println("f1="+f1);//f1=8
        //! true则false false则true
        int g=12;
        System.out.println(!(g>10));
        //^ 相同false不同true
        int g1=12;
        int g2=10;
        System.out.println(g1==12^g2==10);
        System.out.println(g1==12^g2==11);
        //5.位运算符>> << >>> & | ^ ~
        //>>最高位是1则补1，最高位是0则补0
        int h=4;
        int h1=4;
        int h2=4;
        //0100
        System.out.println("h="+(h>>1));
        //0010 2
        //<<左边抛弃，右边补0
        System.out.println("h1="+(h1<<1));
        //1000 8
        //>>>补0
        System.out.println("h2="+(h2>>>1));
        //& | ^两边都是逻辑值时进行逻辑运算，否则进行位运算
        //& 同1则1否则为0
        //| 同0则0 否则为1
        //^ 不同为1相同为0
