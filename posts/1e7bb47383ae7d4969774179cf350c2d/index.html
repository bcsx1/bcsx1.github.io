朴素的模式匹配算法（BF）

      首先展示代码

#include<iostream>
#include<string>
using namespace std;
#define MAXSIZE 20
//命名String为大小是MAXSIZE的字符数组
typedef char String[MAXSIZE];

//返回子串T在主串第pos个字符之后的位置。若不存在，则函数返回0
int Index(String S, String T, int pos)
{
	//初始化i，i表示在主串上遍历的位置
	int i = pos;
	//初始化j，j表示在子串上遍历的位置
	int j = 1;
	//当i与j都没有遍历完串S和T就继续循环
	while (i <= S[0] && j <= T[0])
	{
		//进行比较
		if (S[i] == T[j])
		{
			//主串S下标i处的数据和子串T下标j处的数据相同的情况
			i++;
			j++;
		}
		else
		{
			//主串S下标i处的数据和子串T下标j处的数据不相同的情况
			//i回到上次匹配首位的下一位
			i = i - j + 2;
			j = 1;
		}
	}
	//当是因为遍历到子串的最后一位才退出的循环，就说明找到了子串的位置，其余情况都是没找到
	if (j > T[0])
		//返回i上次匹配的首位
		return i - T[0];
	else
		return 0;
}

//创建一个其值等于s的串S
int StrAssign(String S, char*s)
{
	//串中下标为0的空间储存串的长度
	S[0] = strlen(s);
	//判断字符串s能否储存到串S中
	if (S[0] < MAXSIZE)
	{
		//能储存到串的情况
		for (int i = 1; i <= S[0]; i++)
		{
			//将字符串s的值复制到串S中
			S[i] = *s;
			s++;
		}
		return 1;
	}
	else
		//不能储存到串的情况
		return 0;
}
int main()
{
	//主字符串s
	char s[] = "goodgoogle";
	//子字符串t
	char t[] = "google";
	//创建串S和T
	String S, T;
	//创建串失败直接退出
	if (StrAssign(S, s) == 0)
		return 0;
	if (StrAssign(T, t) == 0)
		return 0;
	int pos = 2;
	cout << "子串t在主串s第pos个字符之后的位置为：" << Index(S, T, pos);
	system("pause");
	return 0;
}

      串的建立就不多做描述朴素的模式匹配算法（BF）主要在函数Index（）中

      朴素的模式匹配算法（BF）大致思路：

      首先让主串的第pos个字符与子串第一个字符进行比较，如果相同子串遍历到下一个字符的同时主串也遍历到下一个字符，直到主串的字符与子串不相同或者遍历完了子串为止，如果是因为遍历完了子串就说明找到了子串在主串中的位置，返回i上次匹配的首位即可。要是因为主串的字符与子串不相同 i（遍历主串的指针）就回到上次匹配的首位的下一位，j（遍历子串的指针）就回到1，从头遍历子串。直到i遍历完主串（说明没找到子串），j遍历完子串（说明找到了子串）为止。
