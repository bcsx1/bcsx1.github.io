			arr[i].f = 1;
			cnt++;
		}else if(num > 0){
			arr[i].a = num;
			arr[i].f = 0;
		}else{
			i--;
			n--;
			flag = 1;
		}
	}
	sort(arr,arr+n,cmp); 
	if(n<k){//如果必须选0
		ans=0; 
	}else if(cnt==n){//如果全是负数 
		if(k&1){//奇数 
			ans = solve(1);
		}else{//偶数 
			ans = solve(0);
		}
	}else if(cnt==0){//如果全是正数 
		ans = solve(0);
	}else{//有正有负
		ll temp = 0;
		 for(ll i = 0; i < k; i++){
		 	if(arr[i].f==1){
		 		temp++;
			 }
		 }
		 if(temp%2==0){//如果前k大的数有偶数各负数 
		 	ans = solve(0);
		 }else{
		 	//找到前k各里面的绝对值最小的负数和后面最大正数交换
			ll q =0,p = 0;
			for(ll i = k-1; i> 0; i--){
				if(arr[i].f==1){
					p = i;
					break;
				}
			} 
			for(ll i = k; i < n; i++){
				if(arr[i].f==0){
					q = i;
					break;
				}
			}
			if(p!=0&&q!=0){
				swap(arr[p],arr[q]);
				ans = solve(0);
				swap(arr[p],arr[q]);	
			}
			//尝试将前k个里面一个最小正数和后面绝对值最大负数交换
			p = 0,q = 0;
			for(ll i = k;i< n;i++)
				if(arr[i].f == 1)
				{
					q = i;
					break;
				}
			for(ll i = k-1;i>0;i--)
				if(arr[i].f == 0)
				{
					p = i;
					break;
				}
			
			if(p!= 0&&q!= 0)
			{
				swap(arr[p],arr[q]);
				ans = max(ans,solve(0));
				swap(arr[p],arr[q]);
			}
			//假如结果仍然小于0，我们只能尝试从最后往前乘了 
			if(ans< 0)
				ans = solve(1); 
		 }
	}
	if(ans< 0){ 
		if(flag)//这时候0派上用场了 
		{
			cout<<0<<endl;
			return 0;
		}
	} 
	cout << ans;
	return 0;	
	
}
