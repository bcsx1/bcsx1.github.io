    do {

            m = 10 * m + c - '0';
        }
        else {
            if (oldc == '+') {
                sum += m;
            }
            else {
                sum -= m;
            }
            m = 0;

            printf("%3d", sum);
        }
    } while (c != '=');
    return 0;
}

2.c=getchar()
3.如果c<='9'&&c>='0',执行m=10*m+c-'0'
4,否则如果oldc=='+',执行sum+=m
5.否则sum-=m
6.m=0
7.oldc=c
8.输出sum的值
9.判断c！='='执行2，3，4，5，6，7，8，9否则转10
