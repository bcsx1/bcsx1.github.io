C语言——航班信息管理系统（双向链表）

        该系统采用双向链表实现，设计效果如下图所示。系统管理的信息有：航班号、出发地、目的地、起飞时间和票价。实现的功能包括录入航班信息、根据航班号删除对应的航班信息、根据航班号修改（更新）该班次的票价、根据航班号查询该航班的信息，以及显示所有的航班信息。

1.  main.c文件

#include "flight.h"

int main()
{
    int n;      //功能选择
    Info_t information;     //航班信息
    char destFlight[16];    //航班号
    float price;        //票价

    //1.创建头结点
    Manager_t *manager= Flight_Create();

    while (1)
    {
        //2.显示系统界面
        Flight_Interface();

        //3.功能（操作）选择
        printf("请根据对应序号选择功能：");
        scanf("%d",&n);

        //4.执行对应的操作
        switch (n)
        {
        case 0:     //退出系统
            printf("已退出系统！！！\n");
            return 0;

        case 1:     //录入航班信息
            printf("请输入航班号：");
            scanf("%s",information.flight_number);
            printf("出发地：");
            scanf("%s",information.departure);
            printf("目的地：");
            scanf("%s",information.destination);
            printf("起飞时间：");
            scanf("%s",information.take_off_time);
            printf("票价：");
            scanf("%f",&information.price);

            Flight_TailInsert(manager,information);
            break;
        
        case 2:     //删除航班信息
            printf("请输入要删除的航班信息的航班号：");
            scanf("%s",destFlight);

            Flight_FlightNumberDel(manager,destFlight);
            break;

        case 3:     //修改航班票价
            printf("请输入要修改票价的航班号：");
            scanf("%s",destFlight);
            printf("更新票价为：");
            scanf("%f",&price);

            Flight_PriceRevise(manager,destFlight,price);
            break;

        case 4:     //查询航班信息
            printf("请输入要查询的航班信息的航班号：");
            scanf("%s",destFlight);

            Flight_FlightNumberFind(manager,destFlight);
            break;
        
        case 5:     //显示所有航班信息
            Flight_print(manager);
            break;
        
        default:
            printf("输入错误，请重新输入！\n");
            break;
        }
    }
    
    return 0;
}

2. flight.c 文件

#include "flight.h"

//创建航班信息表（头结点）
Manager_t * Flight_Create(void)
{
    //1.为头结点申请堆内存
    Manager_t *manager = (Manager_t *)calloc(1,sizeof(Manager_t));
    if(manager == NULL)
    {
        printf("航班信息表创建失败！\n");
        return NULL;
    }

    //2.初始化头结点
    manager->num = 0;
    manager->head = NULL;
    manager->tail = NULL;

    //3.返回头结点的地址
    return manager;
}

//创建新的航班信息
Flight_t * Flight_NewInfo(Info_t info)
{
    //1.为需要存储的新信息申请堆内存
    Flight_t * newInfo = (Flight_t *)calloc(1,sizeof(Flight_t));
    if(newInfo == NULL)
    {
        printf("为新信息申请堆内存失败！\n");
        return NULL;
    }

    //2.初始化，将新信息存入堆空间
    strcpy(newInfo->flight_number , info.flight_number);
    strcpy(newInfo->departure     , info.departure);
    strcpy(newInfo->destination   , info.destination);
    strcpy(newInfo->take_off_time , info.take_off_time);
    newInfo->price = info.price;
    newInfo->prev = NULL;
    newInfo->next = NULL;

    //3.返回存放新信息的地址
    return newInfo;
}

//判断双向链表是否为空
bool Flight_IsEmpty(Manager_t *manager)
{
    return ((manager->num == 0) ? true : false);
}

// 将新的航班信息插入双向链表中（从链表尾部插入）
bool Flight_TailInsert(Manager_t *manager,Info_t info)
{
    //1.创建新的航班信息结点
    Flight_t *newInfo = Flight_NewInfo(info);
    if(newInfo == NULL)
    {
        printf("新的航班信息创建失败！\n");
        return false;
    }

    //2.将新的航班信息结点插入双向链表尾部
    if (Flight_IsEmpty(manager) == true)    //如果当前双向链表为空，插入的结点既作为首结点，又作为尾结点
    {
        manager->head = newInfo;
        manager->tail = newInfo;
    }
    else
    {
        manager->tail->next = newInfo;      //尾结点的后继指针指向插入的新结点
        newInfo->prev = manager->tail;      //新结点的前驱指针指向尾结点
        manager->tail = newInfo;            //更新头结点中记录的尾结点的地址
    }

    //3.更新头结点中记录的信息数量
    manager->num++;
    printf("航班信息录入成功！！！\n");

    return true;
}

//删除指定航班号的航班信息
bool Flight_FlightNumberDel(Manager_t *manager,char *flight_number)
{
    Flight_t *phead = manager->head;
    //1.判断当前链表是否为空
    if(Flight_IsEmpty(manager) == true)
    {
        printf("当前暂无航班信息，删除失败！\n");
        return false;
    }

    //2.遍历双向链表，查找指定的航班号
    while(phead)
    {
        //如果成功找到匹配的航班号，退出循环
        if(strcmp(phead->flight_number,flight_number) == 0)
        {
            break;
        }
        phead = phead->next;
    }

    //3.判断有没有找到匹配的航班号
    if(phead == NULL)
    {
        printf("暂无匹配的航班信息，删除失败！\n");
        return false;
    }
    else
    {
        if(phead == manager->head)          //如果查找到的是首结点
        {
            manager->head = phead->next;    //更新头结点中记录的首结点的地址
            phead->next->prev = NULL;       //要删除的结点的直接后继结点的 prev 指针域指向 NULL
            phead->next = NULL;             //要删除的结点的 next 指针域指向 NULL
        }
        else if (phead ==manager->tail)     //如果查找到的是尾结点
        {
            manager->tail = phead->prev;    //更新头结点中记录的尾结点的地址
            phead->prev->next = NULL;       //要删除的结点的直接前驱结点的 next 指针域指向 NULL
            phead->prev = NULL;             //要删除的结点的 prev 指针域指向 NULL
        }
        else        //查找到的是中间结点
        {
            phead->prev->next = phead->next;    //要删除的结点的直接前驱的 next 指针域指向要删除的结点的直接后继
            phead->next->prev = phead->prev;    //要删除的结点的直接后继的 prev 指针域指向要删除的结点的直接前驱
            phead->prev = NULL;
            phead->next = NULL;
        }
    }

    //4.释放要删除的结点的堆空间
    free(phead);

    //5.更新头结点中记录的信息数量
    manager->num--;
    printf("航班信息删除成功！！！\n");

    return true;
}

//查找并显示指定航班号的航班信息
bool Flight_FlightNumberFind(Manager_t *manager,char *flight_number)
{
    Flight_t *phead = manager->head;
    //1.判断当前链表是否为空
    if(Flight_IsEmpty(manager) == true)
    {
        printf("当前暂无航班信息，查找失败！\n");
        return false;
    }

    //2.遍历双向链表，查找指定的航班号
    while(phead)
    {
        //如果成功找到匹配的航班号，退出循环
        if(strcmp(phead->flight_number,flight_number) == 0)
        {
            break;
        }
        phead = phead->next;
    }

    //3.判断有没有找到匹配的航班号,如果没有找到直接退出函数
    if(phead == NULL)
    {
        printf("暂无匹配的航班信息，查找失败！\n");
        return false;
    }

    //4.找到指定的航班信息，输出显示
    printf("航班号\t\t出发地\t\t目的地\t\t起飞时间\t票价\n");
    printf("%s\t\t%s\t\t%s\t\t%s\t\t%.2f\n",phead->flight_number,
                                    phead->departure,
                                    phead->destination,
                                    phead->take_off_time,
                                    phead->price);
    return true;
}

//修改指定航班号的票价
bool Flight_PriceRevise(Manager_t *manager,char *flight_number,float newPrice)
{
    Flight_t *phead = manager->head;
    //1.判断当前链表是否为空
    if(Flight_IsEmpty(manager) == true)
    {
        printf("当前暂无航班信息，票价修改失败！\n");
        return false;
    }

    //2.遍历双向链表，查找指定的航班号
    while(phead)
    {
        //如果成功找到匹配的航班号，退出循环
        if(strcmp(phead->flight_number,flight_number) == 0)
        {
            break;
        }
        phead = phead->next;
    }

    //3.判断有没有找到匹配的航班号,如果没有找到直接退出函数
    if(phead == NULL)
    {
        printf("暂无匹配的航班信息，票价修改失败！\n");
        return false;
    }

    //4.找到指定航班号的航班信息
    phead->price = newPrice;

    //5.输出提示信息，并将更新后的航班信息显示
    printf("%s次航班票价修改成功!!!\n",phead->flight_number);

    printf("航班号\t\t出发地\t\t目的地\t\t起飞时间\t票价\n");
    printf("%s\t\t%s\t\t%s\t\t%s\t\t%.2f\n",phead->flight_number,
                                    phead->departure,
                                    phead->destination,
                                    phead->take_off_time,
                                    phead->price);

    return true;
}

//打印所有航班信息
bool Flight_print(Manager_t *manager)
{
    Flight_t *phead = manager->head;
    //1.判断当前链表是否为空
    if(Flight_IsEmpty(manager) == true)
    {
        printf("当前暂无航班信息！\n");
        return false;
    }

    //2.打印航班信息表的表头
    printf("航班号\t\t出发地\t\t目的地\t\t起飞时间\t票价\n");

    //3.打印航班信息
    while (phead)
    {
        printf("%s\t\t%s\t\t%s\t\t%s\t\t%.2f\n",phead->flight_number,
                                    phead->departure,
                                    phead->destination,
                                    phead->take_off_time,
                                    phead->price);
        
        phead = phead->next;
    }

    return true;
    
}

#ifndef _FLIGHT_H
#define _FLIGHT_H

#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <stdbool.h>

// 航班信息结构体
typedef struct flight
{
    char flight_number[16]; //航班号
    char departure[32];     //出发地
    char destination[32];   //目的地
    char take_off_time[16]; //起飞时间
    float price;            //价格

    struct flight *prev;     //记录直接前驱的地址
    struct flight *next;     //记录直接后继的地址
}Flight_t;

//管理结构体（头结点）
typedef struct manageStructure
{
    int num;    //记录有多少条信息
    Flight_t *head;     //记录首结点的地址
    Flight_t *tail;     //记录尾结点的地址
}Manager_t;

//信息结构体，用于函数之间传递参数，减少参数数量
typedef struct flightInfo
{
    char flight_number[16]; //航班号
    char departure[32];     //出发地
    char destination[32];   //目的地
    char take_off_time[16]; //起飞时间
    float price;            //价格
}Info_t;

Manager_t * Flight_Create(void);

bool Flight_PriceRevise(Manager_t *manager,char *flight_number,float newPrice);

bool Flight_print(Manager_t *manager);
