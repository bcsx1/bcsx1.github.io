def Msqldb(sql):
    db = pymysql.connect(host = 'localhost', port = 3306, user = 'root', passwd = 'xxx', db='xxx', charset = 'utf8')
    cur = db.cursor()
    cur.execute(sql)
    # 判断sql语句是否select开头
    if sql.strip()[:6].upper() == 'SELECT':
        res = cur.fetchall()
    else:
        db.commit()
        res = 'OK'
    cur.close()
    db.close()
    return res

# 本地数据库插入方法
def insertData(data, table): # data为对应的字典
  db = pymysql.connect(host = 'localhost', port = 3306, user = 'root', passwd = 'xxx', db='xxxx', cursorclass=pymysql.cursors.DictCursor, charset = 'utf8')
  cur = db.cursor()
  # 根据数据库自动配对键值对
  table = table
  keys = ', '.join(data.keys())
  values = ', '.join(['%s'] * len(data))
  sql = 'INSERT INTO {table}({keys}) VALUES ({values})'.format(table=table, keys=keys, values=values)
  try:
    cur.execute(sql, tuple(data.values()))
    # print('Successful')
    res = 'Successful'
    db.commit()
  except:
    # print('Failed')
    db.rollback()
    res = 'Failed'
  cur.close()
  return res
