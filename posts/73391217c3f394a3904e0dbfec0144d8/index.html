SSM框架搭建（IDEA）

一、新建一个Maven项目

打开IDEA软件，File->New->Project ->Maven->Create from archetype

在src下面的main文件夹上右击，New->Directory，新建一个Java目录和一个resources，test目录

 然后在Java目录上右键，Mark Directory as -> Sources Root

接着在resources目录上右键，Mark Directory as -> Resources Root

接着在test目录上右键，Mark Directory as -> Test Sources Root

package com.ssm.entity;

import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@AllArgsConstructor
@NoArgsConstructor
public class Result {
    private int code;   //成功：1   失败：0
    private String message;
    private Object data;

    public static Result getSuccess(Object Data,String message){
        Result result=new Result();
        result.setCode(1);
        result.setMessage(message);
        result.setData(Data);
        return result;
    }

    public static Result getSuccess(){
        return getSuccess(null,null);
    }

    public static Result getFail(String message,Object Data){
        Result result=new Result();
        result.setCode(0);
        result.setMessage(message);
        result.setData(Data);
        return result;
    }

    public static Result getFail(){
        return getFail(null,null);
    }
}

package com.ssm.service.impl;

import com.ssm.entity.User;
import com.ssm.mapper.UserMapper;
import com.ssm.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;


@Service("userService")
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;
    
    //获取所有用户信息
    @Override
    public List<User> getAll() {
        return userMapper.selectList(null);
    }

}

5、创建UserController

package com.ssm.controller;

import com.ssm.entity.Result;
import com.ssm.entity.User;
import com.ssm.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.*;

import java.util.List;


@Controller
@RequestMapping("/user")
public class UserController {

    @Autowired
    private UserService userService;

    //    查找所有用户
    @ResponseBody
    @RequestMapping("/getAll")
    public Result getAll() {
        List<User> userList = userService.getAll();
        return Result.getSuccess(userList, "所有用户查找成功");
    }


}
