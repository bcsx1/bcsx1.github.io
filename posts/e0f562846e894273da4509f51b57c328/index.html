//判断闰年
	public static boolean JudgeRunNian(int year) {
		boolean run = true;
		if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0) {
			System.out.println(year + "是闰年");
			return run == true;
		}
		System.out.println(year + "是平年");
		return run == false;
	}

	//判断空格
	public static int JudgeSpace(int century, int ct) {
		ct = 0;
		switch (century) {
		case 7:
			break;
		case 1:
			ct = 1;
			System.out.print("    \t");
			break;
		case 2:
			ct = 2;
			System.out.print("         \t");
			break;
		case 3:
			ct = 3;
			System.out.print("                  \t");
			break;
		case 4:
			ct = 4;

			break;

			ct = 5;
			System.out.print("                                      \t");
			break;
		case 6:
			ct = 6;
			System.out.print("                                              \t");
			break; 
		}
		return ct;
	}

  • 通过 蔡勒公式 输出给定月份的第一天
  • 通过判断每月第一天,输出相应空格,然后循环往复剩余天数,顺便在天数加的同时,加个判断,输出七天后就得换行
