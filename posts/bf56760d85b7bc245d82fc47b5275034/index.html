部署步骤

准备工作

需安装kubernetes集群；
需安装Nexus3；
需安装consul作为注册中心和配置中心；
需安装postgresql作为高可用分布式事务数据持久化数据库；
需安装NFS持久化插件。
如果未安装参考我的【kubernetes集群部署教程】【kubernetes部署Nexus3教程】【kubernetes部署consul教程】【kubernetes部署postgresql教程】

注意事项

注意1.seata官网给的部署教程http://seata.io/zh-cn/docs/ops/deploy-by-kubernetes.html截止2023年7月适用于低版本的seata。
注意2.在docker的镜像官网https://hub.docker.com/r/seataio/seata-server/tags上查看到seata的镜像版本有下边这些：

注意3.由于我这里使用springboot3.x，所以低版本的seata就不要考虑的，至少要seata1.6+版本才支持，所以这里我选择部署seataio/seata-server:1.7.0这个版本
注意4.网上的绝大多数教程都是采用nacos+mysql来部署seata，我这里是采用consul+postgresql来部署，http://seata.io/zh-cn/docs/user/configuration/consul.html。
注意5.helm的官方镜像里是没有seata这个中间件官方镜像的，所以这我采用原生kubernetes来部署seata，有兴趣的自己制作helm镜像吧。

安装seata集群

# 管理镜像到自己的nexus3私服，如果不采用私服可以不管
docker pull seataio/seata-server:1.7.0
docker tag seataio/seata-server:1.7.0 10.108.69.78:8082/seataio/seata-server:1.7.0
docker push 10.108.69.78:8082/seataio/seata-server:1.7.0

编辑seata-1.7.yaml文件，内容如下：

apiVersion: v1
kind: Service
metadata:
  name: seata-ha-server
  namespace: default
  labels:
    app.kubernetes.io/name: seata-ha-server
spec:
  type: ClusterIP
  # 强制设置ClusterIP，方便管理，生产环境不需要设置
  clusterIP: 10.107.221.112
  ports:
    - port: 8091
      protocol: TCP
      name: http
  selector:
    app.kubernetes.io/name: seata-ha-server

---

apiVersion: apps/v1
kind: Deployment
metadata:
  name: seata-ha-server
  namespace: default
  labels:
    app.kubernetes.io/name: seata-ha-server
spec:
  # 副本数量
  replicas: 3
  selector:
    matchLabels:
      app.kubernetes.io/name: seata-ha-server
  template:
    metadata:
      labels:
        app.kubernetes.io/name: seata-ha-server
    spec:
      containers:
        - name: seata-server
          # 私有仓库镜像，不使用私有仓库则更改为docker.io/seataio/seata-server:1.7.0即可
          image: 10.108.69.78:8083/seataio/seata-server:1.7.0
          imagePullPolicy: IfNotPresent
          ports:
            - name: http
              containerPort: 8091
              protocol: TCP
          volumeMounts:
            - name: seata-config
              # 这里要非常注意，要下载seata1.7的源码和部署包来阅读才知道新版本的配置文件部署在这个目录才能正常加载程序
              mountPath: /seata-server/resources/
      volumes:
        - name: seata-config
          nfs:
            # 下载seata1.7的源码的目录seata-1.7.0\server\src\main\resources的所有文件丢到master节点的/data/nfs/seata-config目录下
            path: /data/nfs/seata-config
            server: 192.168.1.200

然后在postgresql数据库中新建一个名称叫做seata的database，然后创建几个需要的表，建表语句可以在下载的seata启动包中找到

我还是贴一下建表sql语句：

-- -------------------------------- The script used when storeMode is 'db' --------------------------------
-- the table to store GlobalSession data
CREATE TABLE IF NOT EXISTS public.global_table
(
    xid                       VARCHAR(128) NOT NULL,
    transaction_id            BIGINT,
    status                    SMALLINT     NOT NULL,
    application_id            VARCHAR(64),
    transaction_service_group VARCHAR(64),
    transaction_name          VARCHAR(128),
    timeout                   INT,
    begin_time                BIGINT,
    application_data          VARCHAR(2000),
    gmt_create                TIMESTAMP(0),
    gmt_modified              TIMESTAMP(0),
    CONSTRAINT pk_global_table PRIMARY KEY (xid)
);

CREATE INDEX idx_status_gmt_modified ON public.global_table (status, gmt_modified);
CREATE INDEX idx_transaction_id ON public.global_table (transaction_id);


-- the table to store BranchSession data
CREATE TABLE IF NOT EXISTS public.branch_table
(
    branch_id         BIGINT       NOT NULL,
    xid               VARCHAR(128) NOT NULL,
    transaction_id    BIGINT,
    resource_group_id VARCHAR(64),
    resource_id       VARCHAR(256),
    branch_type       VARCHAR(8),
    status            SMALLINT,
    client_id         VARCHAR(64),
    application_data  VARCHAR(2000),
    gmt_create        TIMESTAMP(6),
    gmt_modified      TIMESTAMP(6),
    CONSTRAINT pk_branch_table PRIMARY KEY (branch_id)
);

CREATE SEQUENCE branch_table_seq START 1;
ALTER TABLE public.branch_table  ALTER COLUMN branch_id SET DEFAULT nextval('branch_table_seq');

CREATE INDEX idx_branch_table_xid ON public.branch_table (xid);

-- the table to store lock data
CREATE TABLE IF NOT EXISTS public.lock_table
(
    row_key        VARCHAR(128) NOT NULL,
    xid            VARCHAR(128),
    transaction_id BIGINT,
    branch_id      BIGINT       NOT NULL,
    resource_id    VARCHAR(256),
    table_name     VARCHAR(64),
    pk             VARCHAR(64),
    status         SMALLINT     NOT NULL DEFAULT 0,
    gmt_create     TIMESTAMP(0),
    gmt_modified   TIMESTAMP(0),
    CONSTRAINT pk_lock_table PRIMARY KEY (row_key)
);

comment on column public.lock_table.status is '0:locked ,1:rollbacking';
CREATE INDEX idx_branch_id ON public.lock_table (branch_id);
CREATE INDEX idx_lock_table_xid ON public.lock_table (xid);
CREATE INDEX idx_status ON public.lock_table (status);

CREATE TABLE  public.distributed_lock (
    lock_key     VARCHAR(20)  NOT NULL,
    lock_value        VARCHAR(20)  NOT NULL,
    expire       BIGINT       NOT NULL,
    CONSTRAINT pk_distributed_lock_table PRIMARY KEY (lock_key)
);
