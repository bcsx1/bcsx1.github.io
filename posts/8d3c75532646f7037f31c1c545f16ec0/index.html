D. Distinct Characters Queries(线段树或树状数组)
题意: 给你一个字符串，有两个操作，1.修改某个位置 2.询问某段子串中不同的字符数
思路: 用26颗线段树或者26个树状数组或者26个set记录即可

#include<bits/stdc++.h>
#define lowbit(x) x & -x
using namespace std;
const int N=1e5+5;
char s[N];
int len;
struct node
{
	int c[N];
	void add(int pos,int x)
	{
		while(pos<=len)
		{
			c[pos]+=x;
			pos+=lowbit(pos);
		}
	}
 
	int query(int pos)
	{
		int res=0;
		while(pos>0)
		{
			res+=c[pos];
			pos-=lowbit(pos);
		}
		return res;
	}
}tree[26];
 
int main()
{
	scanf("%s",s+1);
	len=strlen(s+1);
	for(int i=1;i<=len;i++)
		tree[s[i]-'a'].add(i,1);
	int n;
	scanf("%d",&n);
	while(n--)
	{
		int op,l,r;
		scanf("%d%d",&op,&l);
		if(op==1){
			char tmp;
			cin>>tmp;
			tree[s[l]-'a'].add(l,-1);
			s[l]=tmp;
			tree[s[l]-'a'].add(l,1);
		}
		else{
			scanf("%d",&r);
			int ans=0;
			for(int i=0;i<26;i++)
			{
				int tmp=tree[i].query(r)-tree[i].query(l-1);
				if(tmp)
					ans++;
			}
			printf("%d\n",ans);
		}
	}
	return 0;
}

E. Special Permutations(差分前缀和)
题意: 给定一个长度为m的序列x, 排列 p i ( n ) : [ i , 1 , 2 , … , i − 1 , i + 1 , … , n ] p_i(n) : [i,1,2,…,i−1,i+1,…,n] pi​(n):[i,1,2,…,i−1,i+1,…,n]
pos(p,x)表示x在排列p中的第几位
且 f ( p ) = ∑ i = 1 m − 1 ∣ p o s ( p , x i ) − p o s ( p , x i + 1 ) ∣ f(p)=∑_{i=1}^{m−1}|pos(p,x_{i})−pos(p,x_{i+1})| f(p)=∑i=1m−1​∣pos(p,xi​)−pos(p,xi+1​)∣,
求f(p1(n)),f(p2(n)),…,f(pn(n)).
思路: 两种做法:

  1. 差分+前缀和
    对于某一对相邻的数 l = m i n ( x i , x i + 1 ) , r = m a x ( x i , x i + 1 ) l=min(x_{i},x_{i+1}),r=max(x_{i},x_{i+1}) l=min(xi​,xi+1​),r=max(xi​,xi+1​),考虑他们对 p j ( n ) ( j = 1 … … n ) p_j(n)(j=1……n) pj​(n)(j=1……n)的贡献

  2. 因为p1可以很容易得到，所以我们想是否其余的pi由p1转移得到
    我们可以将 x i , x i + 1 x_{i},x_{i+1} xi​,xi+1​看做一条线段(不包含左右端点)，计算每个点被线段覆盖多少次，接着计算每个端点的贡献，每个 x i x_i xi​顶多被算两次，所以可以很容易得出其余 f ( p i ( n ) ) f(p_i(n)) f(pi​(n))

//1.差分+前缀和
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e5+5;
ll cnt[N];
int x[N];
void add(int l,int r,int v)
{
	cnt[l]+=v;
	cnt[r+1]-=v;
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)
		scanf("%d",&x[i]);
	for(int i=1;i<m;i++)
	{
		int l=x[i],r=x[i+1];
		if(l==r)
			continue;
		if(l>r)
			swap(l,r);
		add(1,l-1,r-l);
		add(l,l,r-1);
		add(l+1,r-1,r-l-1);
		add(r,r,l);
		add(r+1,n,r-l);
	}
	for(int i=1;i<=n;i++)
	{
		cnt[i]+=cnt[i-1];
		printf("%lld ",cnt[i]);
	}
	puts("");
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e5+5;
int x[N],cnt[N];
ll res[N];
vector<int>aj[N];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)
		scanf("%d",&x[i]);
	for(int i=1;i<m;i++)
		res[1]+=abs(x[i]-x[i+1]);
	for(int i=1;i<m;i++)
	{
		int l=x[i],r=x[i+1];
		if(l==r)
			continue;
		aj[l].push_back(r),
		aj[r].push_back(l);
		if(r<l)
			swap(l,r);
		if(r-l<2)
			continue;
		cnt[l+1]++;
		cnt[r]--;
	}
	for(int i=1;i<=n;i++)
		cnt[i]+=cnt[i-1];
	for(int i=2;i<=n;i++)
	{
		res[i]=res[1]-cnt[i];
		for(auto j: aj[i])
		{
			res[i]-=abs(j-i);
			res[i]+=j-1+(j<i);
		}
	}
	for(int i=1;i<=n;i++)
		printf("%lld ",res[i]);
	puts("");
	return 0;
}

F. Yet Another Substring Reverse(转化+子集dp)
题意: 给你一个字符串，你可以将某段子串进行翻转，问翻转后最长的一段子串里面全是不相同的字符有多长？只能翻转一次或零次。
思路: 其实这个问题可以转化成求两段不相交子串字符严格不同的长度之和
因为题目说了该字符串只含前20个字符，我们可以进行状压，然后进行转移，先状压每一个严格字符不同子串，然后每个子串作为子集不断转移，最后求每个集合与其补集之和的最大值就是答案

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int dp[1<<20+5],ans;
int main()
{
	string s;
	cin>>s;
	for(int i=0;i<s.size();i++)
	{
		int vis=0;
		for(int j=i;j<s.size();j++)
		{
			int num=s[j]-'a';
			if(vis&(1<<num))
				break;
			vis|=(1<<num);
			dp[vis]=j-i+1;
		}
	}
	for(int i=0;i<20;i++)
	{
		for(int j=0;j<(1<<20);j++)
		{
			if(j&(1<<i))
				dp[j]=max(dp[j],dp[j^(1<<i)]);
		}
	}
	int all=(1<<20)-1;
	for(int i=0;i<(1<<20);i++)
	{
		ans=max(ans,dp[i]+dp[all^i]);
	}
	printf("%d\n",ans);
	return 0;
}
