正整数的因子数和因子和

正整数的因子数和因子和

算术基本定理

对于一个正整数N，可分解成

N= p 1 a 1 p1^{a1} p1a1 p 2 a 2 p2^{a2} p2a2 p 3 a 3 p3^{a3} p3a3… p n a n pn^{an} pnan

那么正因数个数

num=(1+a1)(1+a2)…(1+an)
例如

24= 2 3 ∗ 3 2^3*3 23∗3

num=(3+1)*(1+1)=8

因数，2可以取0,1,2,3个，3可以取0,1个

搭配数4*2

正因数之和

由上知，每个因子大小即 p 1 b 1 p1^{b1} p1b1 p 2 b 2 p2^{b2} p2b2 p 3 b 3 p3^{b3} p3b3… p n b n pn^{bn} pnbn

bn为取的pn的个数

bn可取0,1,2,3…an

因此因数和

sum=(1+p1+ p 1 2 p1^2 p12+ p 1 3 p1^3 p13+…+ p 1 a 1 p1^{a1} p1a1)(1+p2+ p 2 2 p2^2 p22+ p 2 3 p2^3 p23+…+ p 2 a 2 p2^{a2} p2a2)…(1+pn+ p n 2 pn^2 pn2+ p n 3 pn^3 pn3+…+ p n a n pn^{an} pnan)

求因子个数

#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int num,a,m;
	num=1,a=0;//num因子个数，a指数 
	for(int i=2;i*i<=n;i++)
	{
		//一个合数可以由它前面的一个质数乘以一个数得到(素数筛)
		//因此从2开始可以将n变为多个质数的幂乘积。%3后就不可能%9 
		if(n%i==0)
		{   
			a=0;
			while(n%i==0)
			{
				a++;
				n/=i;
			}	
			num*=(1+a);
		}
	}
	if(n>1)num*=2;//n变为质数或本来是质数,相对于a=1,num*=(1+1);  6/2后剩3 
	cout<<num<<endl;
	return 0;
}

求因子和

#include<iostream>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int sum = 1,a;
	for(int i = 2; i * i <= n; i++)
		if(n % i == 0) {
			a = 1;
			while(n % i == 0) a *= i,n /= i;
			sum *= (a * i - 1) / (i - 1);//等比数列求和
		}

	if(n > 1) sum *= (1 + n);//n变为质数或本来是质数
	cout << sum << endl;
	return 0; 
}
