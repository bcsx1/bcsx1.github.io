数据库表创建及所有命令

1.创建数据库，创建英雄表添加对应的约束条件：

#创建数据库
create database if not exists db_zy default charset=utf8mb4;
#创建数据表
create table if not exists t_hero(
	id int primary key auto_increment,
	name varchar(50) not null unique,
	gender enum("男", "女") default "男",
	age int default 18,
	weapon varchar(50) unique
)engine=innodb
