#include "stdio.h"
void sort(int a[],int n)
{
int i,j,t;
for(i=0;i<n-1;i++)
    for(j=0;j<n-1-i;j++)
         if(a[j]>a[j+1])
	{
	t=a[j];a[j]=a[j+1];a[j+1]=t;
	}
}
int main()
{

sort(a,10);
for(i=0;i<10;i++)
