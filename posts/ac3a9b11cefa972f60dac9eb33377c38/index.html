IPC 简介

IPC 是保证多线程之间协助关系和数据传递的一种机制。当系统里有多个任务运行时，经常需要互相无冲突地访问同一个共享资源，比如一些全局变量之类，这时操作系统必须具有对任务运行进行协调的能力，也就是 IPC 机制。

RT-Thread 支持的基础 IPC 有：信号量（sem）、互斥量（mutex）、事件集（event）、邮箱（mailbox）、消息队列（messagequeue）。

源码简析

在此仅对线程间同步部分做简析，即信号量、互斥量、事件集三种。

基础结构

IPC 对象基础数据结构：

struct rt_ipc_object
{
    struct rt_object parent;                            /* 父对象 */

    rt_list_t        suspend_thread;                    /* 挂起线程链表 */
};

rt_ipc_object_init （通过调用 rt_list_init 进行 IPC 链表初始化）：

rt_inline rt_err_t rt_ipc_object_init(struct rt_ipc_object *ipc)
{
    /* IPC 链表初始化 */
    rt_list_init(&(ipc->suspend_thread));

    return RT_EOK;
}

rt_ipc_list_suspend（此函数将挂起 IPC 对象列表的线程。） ：

rt_inline rt_err_t rt_ipc_list_suspend(rt_list_t        *list,
                                       struct rt_thread *thread,
                                       rt_uint8_t        flag)
{
    /* 挂起 IPC 所在线程 */
    rt_thread_suspend(thread);

    /* 判断 IPC 初始化设置的模式 */
    switch (flag)
    {
    /* FIFO（先入先出）模式 */        
    case RT_IPC_FLAG_FIFO:
        /* 将此线程插入挂起链表表尾 */
        rt_list_insert_before(list, &(thread->tlist));
        break; /* RT_IPC_FLAG_FIFO */

    /* PRIO（优先级）模式 */
    case RT_IPC_FLAG_PRIO:
        {
            struct rt_list_node *n;
            struct rt_thread *sthread;

            /* 寻找合适的位置 */
            for (n = list->next; n != list; n = n->next)
            {
                sthread = rt_list_entry(n, struct rt_thread, tlist);//获取线程

                /* 优先级比较 */
                if (thread->current_priority < sthread->current_priority)
                {
                    /* 在 sthread 前插入此线程 */
                    rt_list_insert_before(&(sthread->tlist), &(thread->tlist));
                    break;
                }
            }

            /* 未找到合适位置 */
            if (n == list)
                /* 将此线程插入挂起链表表尾 */
                rt_list_insert_before(list, &(thread->tlist));
        }
        break;/* RT_IPC_FLAG_PRIO */

    default:
        RT_ASSERT(0);
        break;
    }

    return RT_EOK;
}

rt_ipc_list_resume （此函数将唤醒一个线程）:

rt_inline rt_err_t rt_ipc_list_resume(rt_list_t *list)
{
    struct rt_thread *thread;

    /* 获取线程 */
    thread = rt_list_entry(list->next, struct rt_thread, tlist);

    /* 打印日志输出 */
    RT_DEBUG_LOG(RT_DEBUG_IPC, ("resume thread:%s\n", thread->name));

    /* 唤醒线程 */
    rt_thread_resume(thread);

    return RT_EOK;
}

rt_ipc_list_resume_all （此函数将唤醒IPC对象列表中的所有挂起线程，包括IPC对象的挂起列表、邮箱的私有列表等）：

rt_inline rt_err_t rt_ipc_list_resume_all(rt_list_t *list)
{
    struct rt_thread *thread;
    register rt_ubase_t temp;

    /* 唤醒所有挂起的线程 */
    while (!rt_list_isempty(list))
    {
        /* 关中断 */
        temp = rt_hw_interrupt_disable();

        /* 获取下一个挂起的线程 */
        thread = rt_list_entry(list->next, struct rt_thread, tlist);
        /* 将错误码设置为 RT_ERROR */
        thread->error = -RT_ERROR;

        /* 唤醒线程 */
        rt_thread_resume(thread);

        /* 开中断 */
        rt_hw_interrupt_enable(temp);
    }

    return RT_EOK;
}

信号量

信号量是一种轻型的用于解决线程间同步问题的内核对象，线程可以获取或释放它，从而达到同步或互斥的目的。

使用场景：

  • 线程与线程之间的同步
  • 中断与线程之间的同步
  • 资源计数（适合于线程间工作处理速度不匹配的场合）

信号量类型结构：

struct rt_semaphore
{
    struct rt_ipc_object parent;                        /* 父对象 */

    rt_uint16_t          value;                         /* 信号量计数 */
    rt_uint16_t          reserved;                      /* 保留字段 */
};

value 是信号量中的计数值，在信号量被释放时累加，被获取时递减，当其值为 0 ，则再次发出获取请求的线程会被挂起到挂起链表中。

rt_sem_init（信号量初始化）：

rt_err_t rt_sem_init(rt_sem_t       sem,					//信号量对象的句柄
                    const char     *name,					//信号量名称
                    rt_uint32_t    value,					//信号量初始值
                    rt_uint8_t     flag)					//信号量标志
{
    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(value < 0x10000U);

    /* 初始化对象 */
    /* 初始化信号量的内核对象 */
    rt_object_init(&(sem->parent.parent), RT_Object_Class_Semaphore, name);

    /* 初始化信号量的IPC对象 */
    rt_ipc_object_init(&(sem->parent));

    /* 设置信号量计数器的值 */
    sem->value = (rt_uint16_t)value;

    /* 设置信号量的内核对象的标志 */
    sem->parent.parent.flag = flag;

    return RT_EOK;
}

对于静态信号量对象，它的内存空间在编译时期就被编译器分配出来，放在读写数据段或未初始化数据段上，此时使用信号量就不再需要使用 rt_sem_create 接口来创建它，而只需在使用前对它进行初始化即可。当调用这个函数时，系统将对这个 semaphore 对象进行初始化，然后初始化 IPC 对象以及与 semaphore 相关的部分。

rt_sem_create（创建信号量）：

rt_sem_t rt_sem_create(const char *name,					//信号量名称
                        rt_uint32_t value,					//信号量初始值
                        rt_uint8_t flag)					//信号量标志 
{
    rt_sem_t sem;

    RT_DEBUG_NOT_IN_INTERRUPT;//确保此函数未在中断中调用
    RT_ASSERT(value < 0x10000U);//断言检查

    /* 分配对象 */
    /* 动态分配内核对象 */
    sem = (rt_sem_t)rt_object_allocate(RT_Object_Class_Semaphore, name);
    /* 分配失败直接返回 */
    if (sem == RT_NULL)
        return sem;

    /* 初始化信号量的IPC对象 */
    rt_ipc_object_init(&(sem->parent));

    /* 初始化信号量的计数值 */
    sem->value = value;

    /* 设置信号量的内核对象的标志 */
    sem->parent.parent.flag = flag;

    return sem;
}

当调用该接口创建一个信号量时，内核需要先创建一个信号量控制块，当调用上述初始化函数接口时，系统将先从对象管理器中分配一个 semaphore 对象，并初始化这个对象，然后初始化父类 IPC 对象以及与 semaphore 相关的部分。

rt_sem_detach（脱离信号量）：

rt_err_t rt_sem_detach(rt_sem_t sem)						//静态信号量对象的句柄
{
    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(rt_object_get_type(&sem->parent.parent) == RT_Object_Class_Semaphore);
    RT_ASSERT(rt_object_is_systemobject(&sem->parent.parent));

    /* 唤醒信号量内挂载的所有线程 */
    rt_ipc_list_resume_all(&(sem->parent.suspend_thread));

    /* 脱离信号量的内核对象 */
    rt_object_detach(&(sem->parent.parent));

    return RT_EOK;
}

rt_sem_delete（删除信号量）：

rt_err_t rt_sem_delete(rt_sem_t sem)					//rt_sem_create() 创建的信号量对象
{
    RT_DEBUG_NOT_IN_INTERRUPT;//确保此函数未在中断中调用

    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(rt_object_get_type(&sem->parent.parent) == RT_Object_Class_Semaphore);
    RT_ASSERT(rt_object_is_systemobject(&sem->parent.parent) == RT_FALSE);

    /* 唤醒信号量内挂载的所有线程 */
    rt_ipc_list_resume_all(&(sem->parent.suspend_thread));

    /* 删除信号量的内核对象 */
    rt_object_delete(&(sem->parent.parent));

    return RT_EOK;
}

系统不再使用信号量时，可通过删除信号量以释放系统资源，适用于动态创建的信号量。值得注意的是，如果删除该信号量时，有线程正在等待该信号量，那么线程会先唤醒等待在该信号量上的线程，然后再释放信号量的内存资源，原来挂起在信号量上的等待线程将获得 - RT_ERROR 的返回值。

rt_sem_take（等待信号量）：

rt_err_t rt_sem_take(rt_sem_t sem, 							//信号量对象的句柄
                     rt_int32_t time)						//超时时间
{
    register rt_base_t temp;
    struct rt_thread *thread;

    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(rt_object_get_type(&sem->parent.parent) == RT_Object_Class_Semaphore);

    RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(sem->parent.parent)));

    /* 关中断 */
    temp = rt_hw_interrupt_disable();

    /* 打印日志输出 */
    RT_DEBUG_LOG(RT_DEBUG_IPC, ("thread %s take sem:%s, which value is: %d\n",
                                rt_thread_self()->name,
                                ((struct rt_object *)sem)->name,
                                sem->value));
	/* 检查信号量的计数值是否大于0 */
    if (sem->value > 0)
    {
        /* 大于0说明有信号 */
        /* 将计数值减一 */
        sem->value --;

        /* 开中断 */
        rt_hw_interrupt_enable(temp);
    }
    else
    {
        /* 计数值小于等于0，说明没有信号 */
        /* 若等待时间为0，则返回Timeout错误 */
        if (time == 0)
        {
            /* 开中断 */
            rt_hw_interrupt_enable(temp);

            /* 返回超时错误 */
            return -RT_ETIMEOUT;
        }
        else//继续等待信号
        {
            /* 确保未在中断中调用 */
            RT_DEBUG_IN_THREAD_CONTEXT;

            /* 获取当前线程 */
            thread = rt_thread_self();

            /* 设置当前线程的错误代码为RT_EOK */
            thread->error = RT_EOK;

            /* 打印日志输出 */
            RT_DEBUG_LOG(RT_DEBUG_IPC, ("sem take: suspend thread - %s\n",
                                        thread->name));

            /* 挂起当前线程到信号量的挂起链表 */
            rt_ipc_list_suspend(&(sem->parent.suspend_thread),
                                thread,
                                sem->parent.parent.flag);

            /* 有等待时间，则开启线程计时器 */
            if (time > 0)
            {
                /* 打印日志输出 */
                RT_DEBUG_LOG(RT_DEBUG_IPC, ("set thread:%s to timer list\n",
                                            thread->name));

                /* 设置定时器 */
                rt_timer_control(&(thread->thread_timer),
                                 RT_TIMER_CTRL_SET_TIME,
                                 &time);
                /* 启动定时器计时 */
                rt_timer_start(&(thread->thread_timer));
            }

            /* 开中断 */
            rt_hw_interrupt_enable(temp);

            /* 做一次调度 */
            rt_schedule();
	
            /* 若当前线程的错误代码不是RT_EOK，则返回，否则一直阻塞等待信号或超时 */
            if (thread->error != RT_EOK)
            {
                return thread->error;
            }
        }
    }

    RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(sem->parent.parent)));

    return RT_EOK;
}

线程通过获取信号量来获得信号量资源，当信号量值大于零时，线程将获得信号量，并且相应的信号量值会减一。在调用上述接口函数获取资源时，如果信号量的值等于零，则说明当前信号量资源不可用，申请信号量的线程将根据创建 / 初始化时设置的 time 参数的情况选择直接返回、挂起一段时间或永久等待。如果在参数 time 指定的时间内依然得不到信号量，线程将超时返回，返回值是 - RT_ETIMEOUT。

rt_sem_trytake（是rt_sem_take函数的一种特例,时间参数为0）：

rt_err_t rt_sem_trytake(rt_sem_t sem)						//信号量对象的句柄
{
    return rt_sem_take(sem, RT_WAITING_NO);
}

当用户不想在申请的信号量上挂起线程进行等待时，可以使用上述函数接口，进行无等待方式获取信号量，该函数与rt_sem_take(sem, RT_WAITING_NO) 的作用相同，即当线程申请的信号量资源实例不可用的时候，它不会等待在该信号量上，而是直接返回 - RT_ETIMEOUT。

rt_sem_release（释放信号量）：

rt_err_t rt_sem_release(rt_sem_t sem)						//信号量对象的句柄
{
    register rt_base_t temp;
    register rt_bool_t need_schedule;

    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(rt_object_get_type(&sem->parent.parent) == RT_Object_Class_Semaphore);

    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(sem->parent.parent)));

    /* 默认情况下设置调度标志位为不需要重新调度标记 */
    need_schedule = RT_FALSE;

    /* 关中断 */
    temp = rt_hw_interrupt_disable();

    /* 打印日志输出 */
    RT_DEBUG_LOG(RT_DEBUG_IPC, ("thread %s releases sem:%s, which value is: %d\n",
                                rt_thread_self()->name,
                                ((struct rt_object *)sem)->name,
                                sem->value));

    /* 挂起线程不为空 */
    if (!rt_list_isempty(&sem->parent.suspend_thread))
    {
        /* 唤醒第一个挂起的线程 */
        rt_ipc_list_resume(&(sem->parent.suspend_thread));
        /* 设置调度标志位为需要调度 */
        need_schedule = RT_TRUE;
    }
    else//挂起线程为空
    {
        /* 计数量小于最大值 */
        if(sem->value < RT_SEM_VALUE_MAX)
        {
            sem->value ++; /* 计数器加一 */
        }
        else
        {
            rt_hw_interrupt_enable(temp); /* 开中断 */
            return -RT_EFULL; /* 返回RT_EFULL */
        }
    }

    /* 开中断 */
    rt_hw_interrupt_enable(temp);

    /* 判断调度标志位是否需要执行调度 */
    if (need_schedule == RT_TRUE)
        rt_schedule();

    return RT_EOK;
}

释放信号量可以唤醒挂起在该信号量上的线程。当信号量的值等于零，并且有线程等待这个信号量时，释放信号量后将唤醒等待在该信号量线程队列中的第一个线程，由它获取信号量；否则将把信号量的值加 一。

rt_sem_control（信号量控制，只支持重置信号量）：

rt_err_t rt_sem_control(rt_sem_t sem, 					//信号量对象的句柄
                        int cmd, 						//命令（只支持重置信号量）
                        void *arg)
{
    rt_ubase_t level;

    /* 断言检查 */
    RT_ASSERT(sem != RT_NULL);
    RT_ASSERT(rt_object_get_type(&sem->parent.parent) == RT_Object_Class_Semaphore);

    if (cmd == RT_IPC_CMD_RESET)
    {
        rt_ubase_t value;

        /* get value */
        value = (rt_ubase_t)arg;
        /* 关中断 */
        level = rt_hw_interrupt_disable();

        /* 唤醒该信号量上挂起的所有线程 */
        rt_ipc_list_resume_all(&sem->parent.suspend_thread);

        /* 设置计数器值 */
        sem->value = (rt_uint16_t)value;

        /* 开中断 */
        rt_hw_interrupt_enable(level);

        /* 执行一次调度 */
        rt_schedule();

        return RT_EOK;
    }

    return -RT_ERROR;
}

互斥量

互斥量又叫相互排斥的信号量，是一种特殊的二值信号量。互斥量和信号量不同的是：拥有互斥量的线程拥有互斥量的所有权，互斥量支持递归访问且能防止线程优先级翻转；并且互斥量只能由持有线程释放，而信号量则可以由任何线程释放。

互斥量的状态只有两种，开锁或闭锁（两种状态值）。当有线程持有它时，互斥量处于闭锁状态，由这个线程获得它的所有权。相反，当这个线程释放它时，将对互斥量进行开锁，失去它的所有权。

使用场景:

  • 临界资源保护锁（临界资源保护）
  • 具有优先级继承的机制
  • 可防止优先级翻转，避免导致系统死锁

互斥量类型结构：

struct rt_mutex
{
    struct rt_ipc_object parent;                        /* 父对象 */

    rt_uint16_t          value;                         /* 互斥量值 */

    rt_uint8_t           original_priority;             /* 拥有互斥量线程的原优先级 */
    rt_uint8_t           hold;                          /* 线程持有互斥量数量 */

    struct rt_thread    *owner;                         /* 当前拥有互斥量的线程 */
};

rt_mutex_init（互斥量初始化）：

rt_err_t rt_mutex_init (rt_mutex_t mutex,			//互斥量对象的句柄
                        const char* name, 			//互斥量的名称
                        rt_uint8_t flag)			//默认为 RT_IPC_FLAG_PRIO ，无法修改
{
    (void)flag;

    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);

    /* 初始化互斥量对象 */
    rt_object_init(&(mutex->parent.parent), RT_Object_Class_Mutex, name);

    /* 初始化IPC对象 */
    rt_ipc_object_init(&(mutex->parent));

    mutex->value = 1;//设置互斥量值为1
    mutex->owner = RT_NULL;//设置所有者为RT_NULL
    mutex->original_priority = 0xFF;//设置原始优先级为0xFF
    mutex->hold  = 0;//设置所有者数量为0

    /* 标志只能为RT_IPC_flag_PRIO。RT_IPC_FLAG_FIFO无法解决优先级翻转 */
    mutex->parent.parent.flag = RT_IPC_FLAG_PRIO;//设置标志为RT_IPC_FLAG_PRIO

    return RT_EOK;
}

静态互斥量对象的内存是在系统编译时由编译器分配的，一般放于读写数据段或未初始化数据段中。使用该函数接口时，需指定互斥量对象的句柄（即指向互斥量控制块的指针）。

rt_mutex_detach（脱离互斥量）：

rt_err_t rt_mutex_detach(rt_mutex_t mutex)			//互斥量对象的句柄
{
    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mutex->parent.parent) == RT_Object_Class_Mutex);
    RT_ASSERT(rt_object_is_systemobject(&mutex->parent.parent));

    /* 唤醒所有挂起线程 */
    rt_ipc_list_resume_all(&(mutex->parent.suspend_thread));

    /* 脱离互斥量对象 */
    rt_object_detach(&(mutex->parent.parent));

    return RT_EOK;
}

调用该函数接口将把互斥量对象从内核对象管理器中脱离，适用于静态初始化的互斥量。使用该函数接口后，内核先唤醒所有挂在该互斥量上的线程（线程的返回值是 -RT_ERROR），然后系统将该互斥量从内核对象管理器中脱离。

rt_mutex_create（创建互斥量）：

rt_mutex_t rt_mutex_create (const char* name, 		//互斥量的名称
                            rt_uint8_t flag)		//默认为 RT_IPC_FLAG_PRIO ，无法修改
{
    struct rt_mutex *mutex;
    (void)flag;

    RT_DEBUG_NOT_IN_INTERRUPT;//确定不是在中断中使用

    /* 分配互斥量对象 */
    mutex = (rt_mutex_t)rt_object_allocate(RT_Object_Class_Mutex, name);
    /* 分配失败直接返回 */
    if (mutex == RT_NULL)
        return mutex;

    /* 初始化IPC对象 */
    rt_ipc_object_init(&(mutex->parent));

    mutex->value              = 1;//设置互斥量值为1
    mutex->owner              = RT_NULL;//设置所有者为RT_NULL
    mutex->original_priority  = 0xFF;//设置原始优先级为0xff
    mutex->hold               = 0;//设置所有者数量为0

    /* 标志只能为RT_IPC_flag_PRIO。RT_IPC_FLAG_FIFO无法解决优先级翻转 */
    mutex->parent.parent.flag = RT_IPC_FLAG_PRIO;//设置标志为RT_IPC_FLAG_PRIO

    return mutex;
}

可以调用 rt_mutex_create 函数接口动态创建一个互斥量。当调用这个函数时，系统将先从对象管理器中分配一个 mutex 对象，并初始化这个对象，然后初始化父类 IPC 对象以及与 mutex 相关的部分。

rt_mutex_delete（删除互斥量）：

rt_err_t rt_mutex_delete(rt_mutex_t mutex)			//互斥量对象的句柄
{
    RT_DEBUG_NOT_IN_INTERRUPT;//确定不在中断中使用

    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mutex->parent.parent) == RT_Object_Class_Mutex);
    RT_ASSERT(rt_object_is_systemobject(&mutex->parent.parent) == RT_FALSE);

    /* 唤醒所有挂起线程 */
    rt_ipc_list_resume_all(&(mutex->parent.suspend_thread));

    /* 删除互斥量对象 */
    rt_object_delete(&(mutex->parent.parent));

    return RT_EOK;
}

当不再使用互斥量时，通过该函数接口删除互斥量以释放系统资源，适用于动态创建的互斥量。当删除一个互斥量时，所有等待此互斥量的线程都将被唤醒，等待线程获得的返回值是 - RT_ERROR。

rt_mutex_take（获取互斥量）：

rt_err_t rt_mutex_take (rt_mutex_t mutex, 			//互斥量对象的句柄
                        rt_int32_t time)			//指定等待的时间
{
    register rt_base_t temp;
    struct rt_thread *thread;

    /* 确定不是在中断中使用 */
    RT_DEBUG_IN_THREAD_CONTEXT;

    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mutex->parent.parent) == RT_Object_Class_Mutex);

    /* 获取当前线程 */
    thread = rt_thread_self();

    /* 关中断 */
    temp = rt_hw_interrupt_disable();

    //调用回调函数
    RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(mutex->parent.parent)));

    //打印日志输出
    RT_DEBUG_LOG(RT_DEBUG_IPC,
                 ("mutex_take: current thread %s, mutex value: %d, hold: %d\n",
                  thread->name, mutex->value, mutex->hold));

    /* 设置错误码为RT_EOK */
    thread->error = RT_EOK;

    /* 若当前线程已有互斥量 */
    if (mutex->owner == thread)
    {
        /* 拥有互斥量数量小于最大值 */
        if(mutex->hold < RT_MUTEX_HOLD_MAX)
        {
            /* 互斥量hold加一 */
            mutex->hold ++;
        }
        else
        {
            rt_hw_interrupt_enable(temp); /* 开中断 */
            return -RT_EFULL; /* 返回RT_EFULL错误 */
        }
    }
    else
    {
#ifdef RT_USING_SIGNALS
__again:
#endif /* RT_USING_SIGNALS */
        
        /* 互斥量大于0 */
        if (mutex->value > 0)
        {
            /* 互斥量计数值减一 */
            mutex->value --;

            /* 设置互斥量所有者和当前优先级 */
            mutex->owner             = thread;//设置所有线程
            mutex->original_priority = thread->current_priority;//保存当前优先级
            /* 拥有互斥量数量小于最大值 */
            if(mutex->hold < RT_MUTEX_HOLD_MAX)
            {
                mutex->hold ++;//互斥量hold加一
            }
            else
            {
                rt_hw_interrupt_enable(temp); /* 开中断 */
                return -RT_EFULL; /* 返回RT_EFULL错误 */
            }
        }
        else
        {
            /* 不等待，直接返回超时 */
            if (time == 0)
            {
                /* 设置错误值为RT_ETIMEOUT */
                thread->error = -RT_ETIMEOUT;

                /* 开中断 */
                rt_hw_interrupt_enable(temp);

                return -RT_ETIMEOUT;
            }
            else
            {
                /* 打印日志输出 */
                RT_DEBUG_LOG(RT_DEBUG_IPC, ("mutex_take: suspend thread: %s\n",
                                            thread->name));

                /* 更改互斥量所有者线程的优先级 */
                /* 当前线程优先级高于互斥量持有线程的优先级，则改变互斥量优先级为当前线程优先级 */
                if (thread->current_priority < mutex->owner->current_priority)
                {
                    /* 改变持有互斥量线程的优先级 */
                    rt_thread_control(mutex->owner,
                                      RT_THREAD_CTRL_CHANGE_PRIORITY,
                                      &thread->current_priority);
                }

                /* 挂起当前线程 */
                rt_ipc_list_suspend(&(mutex->parent.suspend_thread),
                                    thread,
                                    mutex->parent.parent.flag);

                /* 等待时间未结束则开启计时器 */
                if (time > 0)
                {
                    /* 打印日志输出 */
                    RT_DEBUG_LOG(RT_DEBUG_IPC,
                                 ("mutex_take: start the timer of thread:%s\n",
                                  thread->name));

                    /* 设置线程定时器超时时间 */
                    rt_timer_control(&(thread->thread_timer),
                                     RT_TIMER_CTRL_SET_TIME,
                                     &time);
                    /* 开始计时 */
                    rt_timer_start(&(thread->thread_timer));
                }

                /* 开中断 */
                rt_hw_interrupt_enable(temp);

                /* 做一次调度 */
                rt_schedule();
                
                /* 线程错误码不等于RT_EOK */
                if (thread->error != RT_EOK)
                {
#ifdef RT_USING_SIGNALS
                    /* interrupt by signal, try it again */
                    if (thread->error == -RT_EINTR) goto __again;
#endif /* RT_USING_SIGNALS */

                    /* 返回错误 */
                    return thread->error;
                }
                else
                {
                    /* 关中断 */
                    temp = rt_hw_interrupt_disable();
                }
            }
        }
    }

    /* 开中断 */
    rt_hw_interrupt_enable(temp);

    /* 调用回调函数 */
    RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mutex->parent.parent)));

    return RT_EOK;
}

如果互斥量没有被其他线程控制，那么申请该互斥量的线程将成功获得该互斥量。如果互斥量已经被当前线程线程控制，则该互斥量的持有计数加一，当前线程也不会挂起等待。如果互斥量已经被其他线程占有，则当前线程在该互斥量上挂起等待，直到其他线程释放它或者等待时间超过指定的超时时间。当线程获取了互斥量，那么线程就有了对该互斥量的所有权，即某一个时刻一个互斥量只能被一个线程持有。

rt_mutex_trytake（是rt_mutex_take函数的一种特例,时间参数为0）：

rt_err_t rt_mutex_trytake(rt_mutex_t mutex)					//互斥量对象的句柄
{
    return rt_mutex_take(mutex, RT_WAITING_NO);
}

当用户不想在申请的互斥量上挂起线程进行等待时，可以使用无等待方式获取互斥量。这个函数与 rt_mutex_take(mutex, RT_WAITING_NO) 的作用相同，即当线程申请的互斥量资源实例不可用的时候，它不会等待在该互斥量上，而是直接返回 - RT_ETIMEOUT。

rt_mutex_release（释放互斥量）：

rt_err_t rt_mutex_release(rt_mutex_t mutex)					//互斥量对象的句柄
{
    register rt_base_t temp;
    struct rt_thread *thread;
    rt_bool_t need_schedule;

    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mutex->parent.parent) == RT_Object_Class_Mutex);

    /* 默认情况下设置调度标志位为不需要重新调度标记 */
    need_schedule = RT_FALSE;

    /* 只有线程可以释放互斥量，因为我们需要测试所有权 */
    /* 上下文检查 */
    RT_DEBUG_IN_THREAD_CONTEXT;

    /* 获取当前线程 */
    thread = rt_thread_self();

    /* 关中断 */
    temp = rt_hw_interrupt_disable();

    /* 打印日志输出 */
    RT_DEBUG_LOG(RT_DEBUG_IPC,
                 ("mutex_release:current thread %s, mutex value: %d, hold: %d\n",
                  thread->name, mutex->value, mutex->hold));

    /* 调用回调函数 */
    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mutex->parent.parent)));

    /* 互斥量只能被持有互斥量的线程释放 */
    /* 该线程不是互斥量持有者 */
    if (thread != mutex->owner)
    {
        /* 设置错误码为RT_ERROR */
        thread->error = -RT_ERROR;

        /* 开中断 */
        rt_hw_interrupt_enable(temp);

        /* 返回RT_ERROR */
        return -RT_ERROR;
    }

    /* 互斥量hold减一 */
    mutex->hold --;
    /* 互斥量hold等于零 */
    if (mutex->hold == 0)
    {
        /* 将持有该互斥量的线程的优先级恢复为原来的优先级 */
        if (mutex->original_priority != mutex->owner->current_priority)
        {
            /* 改变为原来的优先级 */
            rt_thread_control(mutex->owner,
                              RT_THREAD_CTRL_CHANGE_PRIORITY,
                              &(mutex->original_priority));
        }

        /* 唤醒挂起的线程 */
        if (!rt_list_isempty(&mutex->parent.suspend_thread))
        {
            /* 获取挂起线程 */
            thread = rt_list_entry(mutex->parent.suspend_thread.next,
                                   struct rt_thread,
                                   tlist);

            /* 打印日志输出 */
            RT_DEBUG_LOG(RT_DEBUG_IPC, ("mutex_release: resume thread: %s\n",
                                        thread->name));

            /* 设置新的线程持有者和优先级 */
            mutex->owner             = thread;//将持有者设置为当前线程
            mutex->original_priority = thread->current_priority;//保存当前优先级

            /* hold不超过最大值 */
            if(mutex->hold < RT_MUTEX_HOLD_MAX)
            {
                /* hold加一 */
                mutex->hold ++;
            }
            else
            {
                rt_hw_interrupt_enable(temp); /* 开中断 */
                return -RT_EFULL; /* 返回RT_EFULL错误 */
            }

            /* 唤醒线程 */
            rt_ipc_list_resume(&(mutex->parent.suspend_thread));

            /* 将标志位设置为需要调度 */
            need_schedule = RT_TRUE;
        }
        else
        {
            /* value不超过最大值 */
            if(mutex->value < RT_MUTEX_VALUE_MAX)
            {
                /* value加一 */
                mutex->value ++;
            }
            else
            {
                rt_hw_interrupt_enable(temp); /* 开中断 */
                return -RT_EFULL; /* 返回RT_EFULL错误 */
            }

            /* 清除持有者 */
            mutex->owner             = RT_NULL;//设置持有者为RT_NULL
            mutex->original_priority = 0xff;//设置原始优先级为0xff
        }
    }

    /* 开中断 */
    rt_hw_interrupt_enable(temp);

    /* 判断是否需要调度 */
    if (need_schedule == RT_TRUE)
        /* 做一次调度 */
        rt_schedule();

    return RT_EOK;
}

当线程完成互斥资源的访问后，应尽快释放它占据的互斥量，使得其他线程能及时获取该互斥量。使用该函数接口时，只有已经拥有互斥量控制权的线程才能释放它，每释放一次该互斥量，它的持有计数就减一。当该互斥量的持有计数为零时（即持有线程已经释放所有的持有操作），它变为可用，等待在该信号量上的线程将被唤醒。如果线程的运行优先级被互斥量提升，那么当互斥量被释放后，线程恢复为持有互斥量前的优先级。

rt_mutex_control（当前没有功能）：

rt_err_t rt_mutex_control(rt_mutex_t mutex, int cmd, void *arg)
{
    /* 断言检查 */
    RT_ASSERT(mutex != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mutex->parent.parent) == RT_Object_Class_Mutex);

    return -RT_ERROR;//返回RT_ERROR
}

事件集

事件集也是线程间同步的机制之一，一个事件集可以包含多个事件，利用事件集可以完成一对多，多对多的线程间同步。事件集主要用于线程间的同步，与信号量不同，它的特点是可以实现一对多，多对多的同步。即一个线程与多个事件的关系可设置为：其中任意一个事件唤醒线程，或几个事件都到达后才唤醒线程进行后续的处理；同样，事件也可以是多个线程同步多个事件。

事件集类型结构：

struct rt_event
{
    struct rt_ipc_object parent;                        /* 父对象 */

    rt_uint32_t          set;                           /* 事件集 */
};

rt_event_init（初始化事件集）：

rt_err_t rt_event_init(rt_event_t event, 			//事件集对象的句柄
                       const char* name, 			//事件集的名称 
                       rt_uint8_t flag);			//事件集的标志
{
    /* 断言检查 */
    RT_ASSERT(event != RT_NULL);

    /* 初始化事件集对象 */
    rt_object_init(&(event->parent.parent), RT_Object_Class_Event, name);

    /* 设置事件集标志 */
    event->parent.parent.flag = flag;

    /* 初始化IPC对象 */
    rt_ipc_object_init(&(event->parent));

    /* 初始化事件集为空 */
    event->set = 0;

    return RT_EOK;
}

静态事件集对象的内存是在系统编译时由编译器分配的，一般放于读写数据段或未初始化数据段中。在使用静态事件集对象前，需要先行对它进行初始化操作。调用该接口时，需指定静态事件集对象的句柄（即指向事件集控制块的指针），然后系统会初始化事件集对象，并加入到系统对象容器中进行管理。

rt_event_detach（脱离事件集）：

rt_err_t rt_event_detach(rt_event_t event)				//事件集对象的句柄
{
    /* 断言检查 */
    RT_ASSERT(event != RT_NULL);
    RT_ASSERT(rt_object_get_type(&event->parent.parent) == RT_Object_Class_Event);
    RT_ASSERT(rt_object_is_systemobject(&event->parent.parent));

    /* 唤醒所有挂起线程 */
    rt_ipc_list_resume_all(&(event->parent.suspend_thread));

    /* 脱离事件集对象 */
    rt_object_detach(&(event->parent.parent));

    return RT_EOK;
}

系统不再使用 rt_event_init() 初始化的事件集对象时，通过脱离事件集对象控制块来释放系统资源。脱离事件集是将事件集对象从内核对象管理器中脱离。用户调用这个函数时，系统首先唤醒所有挂在该事件集等待队列上的线程（线程的返回值是 - RT_ERROR），然后将该事件集从内核对象管理器中脱离。

rt_event_create（创建事件集）：

rt_event_t rt_event_create(const char* name,		//事件集的名称 
                           rt_uint8_t flag);		//事件集的标志
{
    rt_event_t event;

    /* 确保不在中断中使用 */
    RT_DEBUG_NOT_IN_INTERRUPT;

    /* 分配事件集对象 */
    event = (rt_event_t)rt_object_allocate(RT_Object_Class_Event, name);
    
    /* 分配失败 */
    if (event == RT_NULL)
        return event;

    /* 设置事件集标志 */
    event->parent.parent.flag = flag;

    /* 初始化IPC对象 */
    rt_ipc_object_init(&(event->parent));

    /* 初始化事件集为空 */
    event->set = 0;

    return event;
}

当创建一个事件集时，内核首先创建一个事件集控制块，然后对该事件集控制块进行基本的初始化。调用该函数接口时，系统会从对象管理器中分配事件集对象，并初始化这个对象，然后初始化父类 IPC 对象。

rt_event_delete（删除事件集）：

rt_event_send（发送事件可以发送事件集中的一个或多个事件）：

rt_err_t rt_event_send(rt_event_t event, 			//事件集对象的句柄
                       rt_uint32_t set)				//发送的一个或多个事件的标志值
{
    struct rt_list_node *n;
    struct rt_thread *thread;
    register rt_ubase_t level;
    register rt_base_t status;
    rt_bool_t need_schedule;

    /* 断言检查 */
    RT_ASSERT(event != RT_NULL);
    RT_ASSERT(rt_object_get_type(&event->parent.parent) == RT_Object_Class_Event);

    /* 感兴趣的事件为空 */
    if (set == 0)
        return -RT_ERROR;//返回RT_ERROR

    /* 默认设置为不需要调度 */
    need_schedule = RT_FALSE;

    /* 关中断 */
    level = rt_hw_interrupt_disable();

    /* 保存接收到的事件 */
    event->set |= set;

    /* 调用回调函数 */
    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(event->parent.parent)));

    /* 若线程挂起链表不为空 */
    if (!rt_list_isempty(&event->parent.suspend_thread))
    {
        /* 查找并唤醒线程 */
        /* 遍历挂起链表中的所有线程 */
        n = event->parent.suspend_thread.next;
        while (n != &(event->parent.suspend_thread))
        {
            /* 得到挂起线程的线程控制块 */
            thread = rt_list_entry(n, struct rt_thread, tlist);

            /* 将状态设置为RT_ERROR */
            status = -RT_ERROR;
            
            /* 事件集为与事件时 */
            if (thread->event_info & RT_EVENT_FLAG_AND)
            {
                /* 事件集满足所有条件 */
                if ((thread->event_set & event->set) == thread->event_set)
                {
                    /* 将状态设置为RT_EOK */
                    status = RT_EOK;
                }
            }
            /* 事件集为或事件时 */
            else if (thread->event_info & RT_EVENT_FLAG_OR)
            {
                /* 事件集满足部分条件 */
                if (thread->event_set & event->set)
                {
                    /* 保存接收到的事件集 */
                    thread->event_set = thread->event_set & event->set;

                    /* 将状态设置为RT_EOK */
                    status = RT_EOK;
                }
            }
            /* 其他情况 */
            else
            {
                /* 开中断 */
                rt_hw_interrupt_enable(level);

                return -RT_EINVAL;//返回RT_EINVAL错误
            }

            /* 将n指向下一挂起的线程 */
            n = n->next;

            /* 符合条件则唤醒线程 */
            if (status == RT_EOK)
            {
                /* 清除事件 */
                /* 判断是否需要清除接收到的事件 */
                if (thread->event_info & RT_EVENT_FLAG_CLEAR)
                    event->set &= ~thread->event_set;//清除接收到的事件

                /* 唤醒此线程 */
                rt_thread_resume(thread);

                /* 设置为需要调度 */
                need_schedule = RT_TRUE;
            }
        }
    }

    /* 开中断 */
    rt_hw_interrupt_enable(level);

    /* 判断是否需要调度 */
    if (need_schedule == RT_TRUE)
        rt_schedule();//做一次调度

    return RT_EOK;
}

发送事件函数可以发送事件集中的一个或多个事件。使用该函数接口时，通过参数 set 指定的事件标志来设定 event 事件集对象的事件标志值，然后遍历等待在 event 事件集对象上的等待线程链表，判断是否有线程的事件激活要求与当前 event 对象事件标志值匹配，如果有，则唤醒该线程。

rt_event_recv（接收事件）：

rt_err_t rt_event_recv(rt_event_t event,			//事件集对象的句柄
                       rt_uint32_t set,				//接收线程感兴趣的事件
                       rt_uint8_t option,			//接收选项
                       rt_int32_t timeout,			//指定超时时间
                       rt_uint32_t* recved)			//指向接收到的事件
{
    struct rt_thread *thread;
    register rt_ubase_t level;
    register rt_base_t status;

    /* 确保不在中断中使用 */
    RT_DEBUG_IN_THREAD_CONTEXT;

    /* 断言检查 */
    RT_ASSERT(event != RT_NULL);
    RT_ASSERT(rt_object_get_type(&event->parent.parent) == RT_Object_Class_Event);

    /* 感兴趣的事件为空 */
    if (set == 0)
        return -RT_ERROR;//返回RT_ERROR

    /* 默认不需要调度 */
    status = -RT_ERROR;
    /* 获取当前线程 */
    thread = rt_thread_self();
    /* 设置线程错误码为RT_EOK */
    thread->error = RT_EOK;

    /* 调用回调函数 */
    RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(event->parent.parent)));

    /* 关中断 */
    level = rt_hw_interrupt_disable();

    /* 事件集为与事件时 */
    if (option & RT_EVENT_FLAG_AND)
    {
        /* 判断是否接收到所有感兴趣事件 */
        if ((event->set & set) == set)
            status = RT_EOK;//设置状态为RT_EOK
    }
    /* 事件集为或事件时 */
    else if (option & RT_EVENT_FLAG_OR)
    {
        /* 判断是否接收到任意感兴趣事件 */
        if (event->set & set)
            status = RT_EOK;//设置状态为RT_EOK
    }
    /* 其他情况 */
    else
    {
        /* 非以上两种状态会直接导致断言错误 */
        RT_ASSERT(0);
    }

    /* 若接收事件已成功 */
    if (status == RT_EOK)
    {
        /* recved不为空 */
        if (recved)
            *recved = (event->set & set);//保存接收到的事件

        /* 填入线程事件信息 */
        thread->event_set = (event->set & set);//设置线程事件集
        thread->event_info = option;//设置线程事件信息

        /* 是否需要清除事件 */
        if (option & RT_EVENT_FLAG_CLEAR)
            event->set &= ~set;//清除事件
    }
    /* 超时时间为0 */
    else if (timeout == 0)
    {
        /* 设置线程错误码为RT_ETIMEOUT */
        thread->error = -RT_ETIMEOUT;

        /* 开中断 */
        rt_hw_interrupt_enable(level);

        return -RT_ETIMEOUT;//返回RT_ETIMEOUT错误
    }
    else
    {
        /* 填入线程事件信息 */
        thread->event_set  = set;//设置线程事件集
        thread->event_info = option;//设置线程事件信息

        /* 将线程放入挂起链表 */
        rt_ipc_list_suspend(&(event->parent.suspend_thread),
                            thread,
                            event->parent.parent.flag);

        /* 等待时间不为0，启动计时器 */
        if (timeout > 0)
        {
            /* 设置线程定时器超时时间 */
            rt_timer_control(&(thread->thread_timer),
                             RT_TIMER_CTRL_SET_TIME,
                             &timeout);
            /* 启动定时器 */
            rt_timer_start(&(thread->thread_timer));
        }

        /* 开中断 */
        rt_hw_interrupt_enable(level);

        /* 做一次调度 */
        rt_schedule();

        /* 线程错误码不为RT_EOK */
        if (thread->error != RT_EOK)
        {
            /* 返回错误码 */
            return thread->error;
        }

        /* 关中断 */
        level = rt_hw_interrupt_disable();

        /* 保存接收到的事件 */
        if (recved)
            *recved = thread->event_set;
    }

    /* 开中断 */
    rt_hw_interrupt_enable(level);

    /* 调用回调函数 */
    RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(event->parent.parent)));

    return thread->error;//返回线程错误码
}

内核使用 32 位的无符号整数来标识事件集，它的每一位代表一个事件，因此一个事件集对象可同时等待接收 32 个事件，内核可以通过指定选择参数 “逻辑与” 或“逻辑或”来选择如何激活线程，使用 “逻辑与” 参数表示只有当所有等待的事件都发生时才激活线程，而使用 “逻辑或” 参数则表示只要有一个等待的事件发生就激活线程。

当用户调用这个接口时，系统首先根据 set 参数和接收选项 option 来判断它要接收的事件是否发生，如果已经发生，则根据参数 option 上是否设置有 RT_EVENT_FLAG_CLEAR 来决定是否重置事件的相应标志位，然后返回（其中 recved 参数返回接收到的事件）；如果没有发生，则把等待的 set 和 option 参数填入线程本身的结构中，然后把线程挂起在此事件上，直到其等待的事件满足条件或等待时间超过指定的超时时间。如果超时时间设置为零，则表示当线程要接受的事件没有满足其要求时就不等待，而直接返回 - RT_ETIMEOUT。

rt_event_control（事件集控制，只支持重置事件集）：

rt_err_t rt_event_control(rt_event_t event, 		//事件集对象的句柄
                          int cmd, 					//命令（只支持重置信号量）
                          void *arg)
{
    rt_ubase_t level;

    /* 断言检查 */
    RT_ASSERT(event != RT_NULL);
    RT_ASSERT(rt_object_get_type(&event->parent.parent) == RT_Object_Class_Event);

    if (cmd == RT_IPC_CMD_RESET)
    {
        /* 关中断 */
        level = rt_hw_interrupt_disable();

        /* 唤醒所有因事件集等待的线程 */
        rt_ipc_list_resume_all(&event->parent.suspend_thread);

        /* 初始化事件集为空 */
        event->set = 0;

        /* 开中断 */
        rt_hw_interrupt_enable(level);

        /* 开中断 */
        rt_schedule();

        return RT_EOK;
    }

    return -RT_ERROR;
}

补充

优先级翻转问题

优先级翻转指的是当一个高优先级线程试图通过信号量机制访问共享资源时，若该信号量已被低优先级线程所持有，而这个低优先级线程在运行过程中可能又被别的中等优先级线程抢占，从而导致高优先级线程被许多别的低优先级线程所阻塞，难以保证其实时性的情况。

总结来说，就是高优先级线程等待资源的过程中，被其他较低优先级的线程抢占了，造成了低优先级线程“抢占”高优先级线程的现象。

这种情况下就有了互斥量的引入，互斥量能够有效防止优先级翻转问题，通过互斥量的优先级继承机制，在任务调度过程中，低优先级的线程会暂时和高优先级线程保持同一优先级，这样就不会被其他中等优先级的线程所抢占。

生产者消费者模型

生产者消费者问题是一个经典的、多线程同步问题。

假设有两个线程，其中一个为生产者线程，另一个为消费者线程。两个线程共享一块初始为空、固定大小为 n 的缓冲区。

生产者的工作是 “生产” 一段数据，只有缓冲区没满时，生产者才能把消息放入到缓冲区，否则必须等待，如此反复；

与此同时，只有缓冲区非空时，消费者才能从中取出数据，一次消费一段数据，否则必须等待，如此反复。

该问题的核心就是：

  • 要保证不让生产者在缓冲区满时向其中写入数据；
  • 不让消费者在缓冲区空时从中取出数据；

该问题的本质其实就是同步和互斥的问题，要解决就需要解决线程间的同步和互斥。

可以把缓冲区看作一个临界资源保护区，在某一时刻只能允许一个访问者访问，就是某时刻只能允许一个生产者往其中放入数据或是一个消费者从中取出数据，所以在这儿需要解决一个互斥访问的问题。

再者，生产者和消费者又是协作的关系，只有生产者生产后，消费者才能消费，所以还需要解决一个同步问题。

思路如图：

  • 互斥：给予临界资源一个二值的信号量，即互斥量，首先给予初值为 1 ，生产者访问时，先获取了这个量，此时该量的值变为 0 ，消费者就获取不到，这块空间此刻只能由生产者操作，生产者操作完后再释放这个量，此时值变为 1 ，可以由消费者获取，反之消费者使用这块空间时同理。由此实现临界资源的保护。
  • 同步：给与临界资源两个信号量，缓冲区为空的信号量初始值为这块空间的大小，缓冲区满的信号量初始值为 0 ，当生产者生产一个数据放入缓冲区是，缓冲区为空的信号量被生产者获取一次，其数值减一；当消费者从中取出一个数据后，缓冲区为空的信号量被消费者释放一次，其数值加一。对于另一个缓冲区满的信号量，在生产者放入一个数据后就释放一次，其数值加一；在消费者获取一个数据后被获取一次，其数值减一。当缓冲区满后，生产者获取不到缓冲区为空的信号量，于是不进行写入操作；反之当缓冲区空时，消费者获取不到缓冲区满的信号量，于是不进行读取操作。从而实现了两个线程的同步。

在上述问题中的信号量和互斥量均为独立的模型，仅为说明问题，并非操作系统实际使用中的信号量和互斥量。
