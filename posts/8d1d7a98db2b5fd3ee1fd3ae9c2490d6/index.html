Redis 5.0数据结构之压缩列表ziplist源码详解（二）

由于篇幅过长，ziplist的源码分析分为两篇，本篇是第二篇，涉及到ziplist的添加元素、查找元素、删除元素以及连锁更新实现的源码分析。
Redis 5.0数据结构之压缩列表ziplist源码详解（一）

前文已经提及了压缩列表ziplist的主要设计初衷是尽量节约空间，因此设计出了一系列的压缩编码，将ziplist设计成了内存地址连续，使用基地址+偏移量的方式来访问压缩列表内节点，并且说明了压缩列表节点结构具有的属性以及相应的作用，通过源码分析了对于节点内各属性的编码的实现细节。

接下来将继续通过源码解析ziplist的各操作是如何实现的。

添加元素

添加元素是调用ziplistPush函数，其中参数where参数决定新元素插入位置。

unsigned char *ziplistPush(unsigned char *zl, unsigned char *s, unsigned int slen, int where) {
    unsigned char *p;
    p = (where == ZIPLIST_HEAD) ? ZIPLIST_ENTRY_HEAD(zl) : ZIPLIST_ENTRY_END(zl);
    return __ziplistInsert(zl,p,s,slen);
}

实际完成添加元素工作的是__ziplistInsert函数，源码如下：

// 将值长度为slen的节点存放到zl压缩列表的位置p处
unsigned char *__ziplistInsert(unsigned char *zl, unsigned char *p, unsigned char *s, unsigned int slen) {
	// 因为涉及到位运算 大小端转换
	// Redis默认小端存储 如果在大端存储的机器上允许 需要大小端转换 
	// reqlen为当前节点所需长度
    size_t curlen = intrev32ifbe(ZIPLIST_BYTES(zl)), reqlen;
    unsigned int prevlensize, prevlen = 0;
    // 偏移量
    size_t offset;
    int nextdiff = 0;
    unsigned char encoding = 0;
    long long value = 123456789; /* initialized to avoid warning. Using a value
                                    that is easy to see if for some reason
                                    we use it uninitialized. */
    zlentry tail;

    /* Find out prevlen for the entry that is inserted. */
    // 非尾部插入
    if (p[0] != ZIP_END) {
    	// 获取prevlen 宏定义函数见下
        ZIP_DECODE_PREVLEN(p, prevlensize, prevlen);
    // 尾部插入
    } else {
    	// 获取尾节点
        unsigned char *ptail = ZIPLIST_ENTRY_TAIL(zl);
        // 如果ziplist不为空
        if (ptail[0] != ZIP_END) {
        	// 则prevlen就是尾节点的长度
            prevlen = zipRawEntryLength(ptail);
        }
    }

    /* See if the entry can be encoded */
    // 尝试对value整数编码
    if (zipTryEncoding(s,slen,&value,&encoding)) {
        /* 'encoding' is set to the appropriate integer encoding */
        // 记录整数编码所需长度
        reqlen = zipIntSize(encoding);
    // 没法整数编码 则value保存的是字符串
    } else {
        /* 'encoding' is untouched, however zipStoreEntryEncoding will use the
         * string length to figure out how to encode it. */
        // 直接记录字节数组编码所需长度
        reqlen = slen;
    }
    /* We need space for both the length of the previous entry and
     * the length of the payload. */
    // 需要空间存放上一个节点的长度和当前节点长度
    // 分别获取并累加到reqlen中
    reqlen += zipStorePrevEntryLength(NULL,prevlen);
    reqlen += zipStoreEntryEncoding(NULL,encoding,slen);

    /* When the insert position is not equal to the tail, we need to
     * make sure that the next entry can hold this entry's length in
     * its prevlen field. */
    // forcelarge用于判断prevrawlen编码时需要使用哪个函数处理
    // forcelarge为0时 使用zipStorePrevEntryLength编码 该函数用于处理len小于254的情况
  	// forcelarge为1时 使用zipStorePrevEntryLengthLarge编码 该函数处理len大于等于254的情况  
    int forcelarge = 0;
    // 由于是插入操作 相当于新节点的下一个节点的前置节点更换了 需要判断下一个节点的prevlen是否够用
    // prevlen的值只有两种可能 分别是1或5 zipPrevLenByteDiff函数见下文注释
    nextdiff = (p[0] != ZIP_END) ? zipPrevLenByteDiff(p,reqlen) : 0;
    // 此时有两种情况需要改变 即新节点len为1 下一个节点prevlen为5 或者相反
    // 判断成立的情况是 新节点len为1 下一个节点prevlen为5
    // 需要记录使用zipStorePrevEntryLengthLarge存储prevlen 并且记录nextdiff不需要扩容
    if (nextdiff == -4 && reqlen < 4) {
        nextdiff = 0;
        forcelarge = 1;
    }

    /* Store offset because a realloc may change the address of zl. */
    // 保存偏移量 因为realloc也可能改变ziplist的地址
    offset = p-zl;
    // 重设ziplist的大小
    zl = ziplistResize(zl,curlen+reqlen+nextdiff);
    p = zl+offset;

    /* Apply memory move when necessary and update tail offset. */
    // 如果不是在尾部插入
    if (p[0] != ZIP_END) {
        /* Subtract one because of the ZIP_END bytes */
        // 通过内存拷贝 使得原有数据后移
        memmove(p+reqlen,p-nextdiff,curlen-offset-1+nextdiff);

        /* Encode this entry's raw length in the next entry. */
        // 虽然prev空间有富余 但不进行压缩 防止连锁更新
        if (forcelarge)
            zipStorePrevEntryLengthLarge(p+reqlen,reqlen);
        else
            zipStorePrevEntryLength(p+reqlen,reqlen);

        /* Update offset for tail */
        // 更新ziplist中的尾节点偏移量
        ZIPLIST_TAIL_OFFSET(zl) =
            intrev32ifbe(intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))+reqlen);

        /* When the tail contains more than one entry, we need to take
         * "nextdiff" in account as well. Otherwise, a change in the
         * size of prevlen doesn't have an effect on the *tail* offset. */
        zipEntry(p+reqlen, &tail);
        // 新节点的下一个节点扩展时 需要在ziplist的尾节点偏移量上再加上nextdiff
        if (p[reqlen+tail.headersize+tail.len] != ZIP_END) {
            ZIPLIST_TAIL_OFFSET(zl) =
                intrev32ifbe(intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))+nextdiff);
        }
    } else {
    	// 在尾部插入时 直接修改ziplist的尾节点偏移量
        /* This element will be the new tail. */
        ZIPLIST_TAIL_OFFSET(zl) = intrev32ifbe(p-zl);
    }

    /* When nextdiff != 0, the raw length of the next entry has changed, so
     * we need to cascade the update throughout the ziplist */
   	// 下一个节点需要扩展 触发连锁更新
    if (nextdiff != 0) {
    	// 与之前相同 记录偏移量防止地址改变
        offset = p-zl;
        // 连锁更新 见下文
        zl = __ziplistCascadeUpdate(zl,p+reqlen);
        p = zl+offset;
    }

    /* Write the entry */
    // 节点记录prevlen encoding和len
    p += zipStorePrevEntryLength(p,prevlen);
    p += zipStoreEntryEncoding(p,encoding,slen);
    // 根据不同类型进行数据存储
    if (ZIP_IS_STR(encoding)) {
        memcpy(p,s,slen);
    } else {
        zipSaveInteger(p,value,encoding);
    }
    // ziplist长度+1
    ZIPLIST_INCR_LENGTH(zl,1);
    return zl;
}

插入函数中调用到的函数如下（源码解析一中出现的方法不再重复粘贴）：

// 返回前一个元素的长度 以及用于编码前一个元素长度的字节数
#define ZIP_DECODE_PREVLEN(ptr, prevlensize, prevlen) do {                     \
    ZIP_DECODE_PREVLENSIZE(ptr, prevlensize);                                  \
    if ((prevlensize) == 1) {                                                  \
        (prevlen) = (ptr)[0];                                                  \
    } else if ((prevlensize) == 5) {                                           \
        assert(sizeof((prevlen)) == 4);                                    \
        memcpy(&(prevlen), ((char*)(ptr)) + 1, 4);                             \
        memrev32ifbe(&prevlen);                                                \
    }                                                                          \
} while(0);

// 返回节点的长度
unsigned int zipRawEntryLength(unsigned char *p) {
    unsigned int prevlensize, encoding, lensize, len;
    ZIP_DECODE_PREVLENSIZE(p, prevlensize);
    ZIP_DECODE_LENGTH(p + prevlensize, encoding, lensize, len);
    return prevlensize + lensize + len;
}

// 返回整数编码所需字节
unsigned int zipIntSize(unsigned char encoding) {
    switch(encoding) {
    case ZIP_INT_8B:  return 1;
    case ZIP_INT_16B: return 2;
    case ZIP_INT_24B: return 3;
    case ZIP_INT_32B: return 4;
    case ZIP_INT_64B: return 8;
    }
    if (encoding >= ZIP_INT_IMM_MIN && encoding <= ZIP_INT_IMM_MAX)
        return 0; /* 4 bit immediate */
    panic("Invalid integer encoding 0x%02X", encoding);
    return 0;
}

// 该方法返回编码prevlen所需字节差异
// 如果返回正值说明空间需要扩大 返回负值说明空间有富余 返回0说明空间正好
int zipPrevLenByteDiff(unsigned char *p, unsigned int len) {
    unsigned int prevlensize;
    ZIP_DECODE_PREVLENSIZE(p, prevlensize);
    return zipStorePrevEntryLength(NULL, len) - prevlensize;
}
// 获取prevlensize
#define ZIP_DECODE_PREVLENSIZE(ptr, prevlensize) do {                          \
    if ((ptr)[0] < ZIP_BIG_PREVLEN) {                                          \
        (prevlensize) = 1;                                                     \
    } else {                                                                   \
        (prevlensize) = 5;                                                     \
    }                                                                          \
} while(0);

// 重设ziplist的大小
unsigned char *ziplistResize(unsigned char *zl, size_t len) {
    assert(len < UINT32_MAX);
    zl = zrealloc(zl,len);
    ZIPLIST_BYTES(zl) = intrev32ifbe(len);
    zl[len-1] = ZIP_END;
    return zl;
}

插入流程总结

由于ziplist的实现思想是基地址加偏移量，并且在每个节点中都记录了上一个节点的长度，因此在有新节点插入到ziplist中时，需要关注两件事：

  1. ziplist中的尾偏移量必然修改。ziplist中添加的新节点，则ziplist头举例尾节点的长度一定发生变化。
  2. 新节点的下一个节点的prevlen属性可能修改。新节点插入后相当于下一个节点的前置节点发生改变，而改变前后两节点的长度有可能发生变化。

ziplist插入流程可以总结如下：

  1. 获取当前p指针的prevlen，即插入后新节点的prevlen。
  2. 对value数据进行编码。
  3. 计算新节点需要的长度并记录到reqlen中。
  4. 由于p指针指向节点的前置节点将会改变，需要判断p指针指向节点的prevlen是否足够容纳reqlen，如果不够需要进行扩展，如有富余也不压缩（防止连锁更新）。
  5. 为ziplist重新分配内存，如果新节点不是在尾部插入需要将新节点后的数据进行后移。
  6. 更新ziplist中尾节点偏移量。
  7. 如果p指针指向节点扩容了，则需要进行连锁更新。
  8. 保存当前节点的prevlen、encoding以及其存放的value等。
  9. ziplist长度加1

ziplist的使用场景中大多是头插入或尾插入，因此基地址加偏移量结构带来的中间节点插入时间复杂度较高是可以容忍的。

查找元素

使用ziplistFind从指定位置开始向后查找指定元素。源码如下：

// 参数p为指定起始位置 skip为每次查找跳过的节点 找不到指定元素时返回NULL
unsigned char *ziplistFind(unsigned char *p, unsigned char *vstr, unsigned int vlen, unsigned int skip) {
    int skipcnt = 0;
    unsigned char vencoding = 0;
    long long vll = 0;
	// 一直查找到ziplist尾部
    while (p[0] != ZIP_END) {
        unsigned int prevlensize, encoding, lensize, len;
        unsigned char *q;

		// 记录当前位置 用于移位
        ZIP_DECODE_PREVLENSIZE(p, prevlensize);
        ZIP_DECODE_LENGTH(p + prevlensize, encoding, lensize, len);
        q = p + prevlensize + lensize;

        if (skipcnt == 0) {
            /* Compare current entry with specified entry */
            // 如果是字符串编码
            if (ZIP_IS_STR(encoding)) {
                if (len == vlen && memcmp(q, vstr, vlen) == 0) {
                    return p;
                }
            // 如果是整数编码
            } else {
                /* Find out if the searched field can be encoded. Note that
                 * we do it only the first time, once done vencoding is set
                 * to non-zero and vll is set to the integer value. */
                // 先按照整型进行比较
                if (vencoding == 0) {
                    if (!zipTryEncoding(vstr, vlen, &vll, &vencoding)) {
                        /* If the entry can't be encoded we set it to
                         * UCHAR_MAX so that we don't retry again the next
                         * time. */
                        // 无法进行整数编码 vencoding设为UCHAR_MAX 不再进行整数比较尝试
                        vencoding = UCHAR_MAX;
                    }
                    /* Must be non-zero by now */
                    // 断言函数 vencoding一定不为0
                    assert(vencoding);
                }

                /* Compare current entry with specified entry, do it only
                 * if vencoding != UCHAR_MAX because if there is no encoding
                 * possible for the field it can't be a valid integer. */
                // 如果是整型编码 直接进行比较
                if (vencoding != UCHAR_MAX) {
                    long long ll = zipLoadInteger(q, encoding);
                    if (ll == vll) {
                        return p;
                    }
                }
            }

            /* Reset skip count */
            // 重置跳过节点
            skipcnt = skip;
        } else {
            /* Skip entry */
            // 跳过节点
            skipcnt--;
        }

        /* Move to next entry */
        // 移到下一个节点
        p = q + len;
    }

    return NULL;
}

删除元素

使用__ziplistDelete函数可以从指定位置开始删除若干的节点。源码如下：

// 从p指针的位置开始删除num个节点 最终返回ziplist指针
unsigned char *__ziplistDelete(unsigned char *zl, unsigned char *p, unsigned int num) {
    unsigned int i, totlen, deleted = 0;
    size_t offset;
    int nextdiff = 0;
    zlentry first, tail;

    zipEntry(p, &first);
    for (i = 0; p[0] != ZIP_END && i < num; i++) {
        p += zipRawEntryLength(p);
        deleted++;
    }
	// 记录要删除节点锁占用的字节数
    totlen = p-first.p;
    if (totlen > 0) {
    	// 如果删除的元素没有到达ziplist尾节点	
        if (p[0] != ZIP_END) {
            // 与添加节点方法一样 由于删除节点的下一个节点的前置节点发生了变化
            // 需要判断下一个节点的prevlen是否够用
            nextdiff = zipPrevLenByteDiff(p,first.prevrawlen);
            p -= nextdiff;
            zipStorePrevEntryLength(p,first.prevrawlen);

            /* Update offset for tail */
            // 更新ziplist中的尾节点偏移量
            ZIPLIST_TAIL_OFFSET(zl) =
                intrev32ifbe(intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))-totlen);

            // 需要在ziplist的尾节点偏移量尚再加上nextdiff
            zipEntry(p, &tail);
            if (p[tail.headersize+tail.len] != ZIP_END) {
                ZIPLIST_TAIL_OFFSET(zl) =
                   intrev32ifbe(intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))+nextdiff);
            }

            /* Move tail to the front of the ziplist */
            // 将节点移动到删除元素的位置
            memmove(first.p,p,
                intrev32ifbe(ZIPLIST_BYTES(zl))-(p-zl)-1);
        } else {
            // 如果删除到了尾节点 则不需要进行节点移动
            ZIPLIST_TAIL_OFFSET(zl) =
                intrev32ifbe((first.p-zl)-first.prevrawlen);
        }

        /* Resize and update length */
        // 重新分配ziplist内存空间
        offset = first.p-zl;
        zl = ziplistResize(zl, intrev32ifbe(ZIPLIST_BYTES(zl))-totlen+nextdiff);
        ZIPLIST_INCR_LENGTH(zl,-deleted);
        p = zl+offset;

        /* When nextdiff != 0, the raw length of the next entry has changed, so
         * we need to cascade the update throughout the ziplist */
        // 当nextdiff不为0时 说明删除节点的下一个节点的prevlen需要扩容 触发连锁更新
        if (nextdiff != 0)
            zl = __ziplistCascadeUpdate(zl,p);
    }
    // 返回ziplist
    return zl;
}

连锁更新

前文中已经提及，ziplist中的每一个节点都会使用prevlen属性记录上一个节点的长度，为了节约内存空间，prevlen属性有两种可能的值，分别为1或5。

本文中分析的增加和删除方法中都出现了连锁更新的身影，因为任何修改方法都可能改变节点本身的长度，相当于修改了被修改节点的下一个节点的prevlen属性。

任何修改操作都可能引发prevlen越界问题，即prevlen占据1字节，但其前一个节点修改后长度大于等于了254，无法只用1字节记录，此时就需要修改节点的prevlen，修改prevlen后，当前节点的长度也可能变得大于等于254，导致其下一个节点也需要修改prevlen，故该操作名为连锁更新，最坏的情况整个ziplist修改位置后全部节点全要更新。

由于连锁更新比较耗费资源，所以开发时采取的策略是尽量避免连锁更新，解释了为什么增加元素时，如果prevlen有富余也不进行压缩。

unsigned char *__ziplistCascadeUpdate(unsigned char *zl, unsigned char *p) {
    size_t curlen = intrev32ifbe(ZIPLIST_BYTES(zl)), rawlen, rawlensize;
    size_t offset, noffset, extra;
    unsigned char *np;
    zlentry cur, next;

	// 对ziplist中的所有节点进行遍历
    while (p[0] != ZIP_END) {
    	// 获取当前节点与当前节点长度
        zipEntry(p, &cur);
        rawlen = cur.headersize + cur.len;
        // 获取当前节点prevlen大小
        rawlensize = zipStorePrevEntryLength(NULL,rawlen);

        /* Abort if there is no next entry. */
        // 没有后继节点时 退出函数
        if (p[rawlen] == ZIP_END) break;
        // 获取前驱节点
        zipEntry(p+rawlen, &next);

        /* Abort when "prevlen" has not changed. */
        // prevlen不需要改变 则直接退出函数
        if (next.prevrawlen == rawlen) break;

		// 只有需要扩展时继续连锁更新 从不进行压缩操作
        if (next.prevrawlensize < rawlensize) {
            /* The "prevlen" field of "next" needs more bytes to hold
             * the raw length of "cur". */
            // 与之前相同 记录偏移量防止地址改变
            offset = p-zl;
            // 需要额外开辟的空间
            extra = rawlensize-next.prevrawlensize;
            // 为ziplist重新分配内存
            zl = ziplistResize(zl,curlen+extra);
            p = zl+offset;

            /* Current pointer and offset for next element. */
            np = p+rawlen;
            noffset = np-zl;
			
            /* Update tail offset when next element is not the tail element. */
            // 下一个节点不是尾节点时 ziplist的尾节点偏移量要加上extra 即扩容大小
            if ((zl+intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))) != np) {
                ZIPLIST_TAIL_OFFSET(zl) =
                    intrev32ifbe(intrev32ifbe(ZIPLIST_TAIL_OFFSET(zl))+extra);
            }

            /* Move the tail to the back. */
            // 通过内存拷贝 使得原有数据后移
            memmove(np+rawlensize,
                np+next.prevrawlensize,
                curlen-noffset-next.prevrawlensize-1);
            zipStorePrevEntryLength(np,rawlen);

            /* Advance the cursor */
            // 移动指针 向后检查
            p += rawlen;
            curlen += extra;
        } else {
        	// 对于prevlen大于等于当前所需空间时 直接修改值
        	// 值修改后直接退出函数 避免后续可能的连锁更新 提升效率
            if (next.prevrawlensize > rawlensize) {
                /* This would result in shrinking, which we want to avoid.
                 * So, set "rawlen" in the available bytes. */
                zipStorePrevEntryLengthLarge(p+rawlen,rawlen);
            } else {
                zipStorePrevEntryLength(p+rawlen,rawlen);
            }

            /* Stop here, as the raw length of "next" has not changed. */
            break;
        }
    }
    return zl;
}

总结

Redis3.2之前，当同时满足以下两个条件时，列表键的底层实现为ziplist，其余情况均使用linkedlist

  • 哈希对象保存的所有键值的字符串长度小于64字节；
  • 哈希对象保存的键值对数量小于512个；

由使用ziplist的场景可见，ziplist的设计初衷是为了尽可能的节约内存空间，提升存储效率。

在结构设计上，ziplist虽然叫做list，但与传统链表不同，数据结构上没有采取直接指针指向实现，而是采取了连续内存地址+偏移量的方式实现。ziplist中的每个entry节点都存储着前驱节点的长度以及自身的长度，这样的属性设计决定了ziplist可以从头到尾或者从尾到头进行遍历。

在数据存储上，可以在同一个ziplist中存储字符串或整数，对于不同长度的字符串或整数，ziplist采取不同的编码形式，最大程度的压缩使用空间。

在对ziplist修改时，有可能由于节点长度的改变引发连锁更新，最坏的情况会更新整个ziplsit，对于连锁更新，Redis设计的策略是能避免则避免，提升效率。
