一、运算符重载

a、什么是运算符重载？
定义：运算重载是赋予自定义数据能被运算符直接操作，运算符重载实质是函数调用。

如：+运算符，一般情况下只能运算两个int | float | double 的数据，但如果我们将两个字符串加在一起，就需要对+运算符进行重载。

注意：一般我们将两个字符串相加只需要包含#include头文件即可，这里头文件中已经为我们封装好了+运算符重载。

b、运算符重载中的一些名词

  • 函数返回值类型： 这个运算符最终运算符结果的类型 MM a MM b MM c=a+b c 就是为MM

  • 函数名: operator加上运算符组成函数名 operator+ 这个整体函数名

  • 函数参数: 看运算符需要几个参数去运算，传几个参数

    • 友元重载: 参数个数是等于操作数
    • 类重载： 参数个数是等于操作数-1

  • 参数类型： 操作数的类型决定

  • 函数体： 写你想要的运算(不能违背运算符本来含义)

c、运算符重载的一般格式

一般函数重载格式：函数返回值类型 + 函数名 (函数参数) { 函数体 }
友元函数重载格式：friend + 函数返回值类型 + 函数名 (函数参数) { 函数体 }

例如：//友元重载 girl + mm
friend MM operator+(MM& object1, MM& object2);

d、运算符重载的显隐式调用
一般使用的是隐式调用，但代码底层还是将其解析为显式调用格式，再去调用。

例如：重载了MM类中的+运算符
MM result = girl + mm; -------隐式调用
operator+(girl,mm); -------显式调用
注意：无论是显式还是隐式调用，效果都是一样的。

class MM 
{
public:
	MM() {}
	MM(string name,int age):name(name),age(age){}
	void printMM() 
	{
		cout << name << "\t" << age << endl;
	}
	//友元重载  girl + mm
	friend MM operator+(MM& object1, MM& object2);
	
private:
	string name;
	int age;
};
MM operator+(MM& object1, MM& object2) 
{
	//return MM(object1.name, object1.age + object2.age);
	MM mm;
	mm.name = object1.name;
	mm.age = object1.age + object2.age;
	return mm;
}

int main(){
	MM girl("小乔", 10);
	MM mm("小乔", 28);
	//vs中重载的运算符是绿色
	MM result = girl + mm;    //运算符重载隐式调用 //girl+mm解析过程: operator+(girl,mm);
}

1.1、友元重载

C++中的友元部分：https://blog.csdn.net/weixin_64603510/article/details/126788489

友元重载与普通运算符重载的区别：

1、友元重载比普通函数重载在声名时多了一个friend关键字。
2、友元重载括号中的参数个数=操作数 - 1
3、友元重载中可以使用this关键字
4、友元函数重载也相当是于友元函数，一样可以忽视类中的权限限制。
注意：友元函数重载(以及友元函数) 可使用传参 和 不传参两种方式，但在友元函数重载中一般推荐使用传参的方式。

#include <iostream>

//正常友元函数的写法
using namespace std;
class MM 
{
public:
	MM(string name = "默认", int age = 12) :name(name), age(age) {}
	//友元函数
	friend void print(MM& object);
	void printMM() 
	{
		cout << name << "\t" << age << endl;
	}
private:
	string name;
	int age;
};
//类外实现不需要类名限定
void print(MM& object)
{
	MM* p = new MM;
	p->name = "无视权限";
	p->age = 123;
	p->printMM();
	cout << object.name << "\t" << object.age << endl;
}

//不正常的写法: 以另一个类的成员函数当做友元函数
class B
{
public:
	B(int b = 1234) :b(b) {}
	void printB();

private:
	int b;
};
class A 
{
	friend void B::printB();		//不受A的权限先定义
public:
	A(int a = 122) :a(a) {}
private:
	int a;
};
void B::printB() 
{
	A  aobject;
	cout << aobject.a << endl;
	cout << "b中的函数" << endl;
}
int main() 
{
	MM  mm;
	//cout << mm.name << endl;  //错误 没有权限
	print(mm);
	B bObject;
	A aObject;
	bObject.printB();
	return 0;
}

1.2 类重载

定义:在另外一个用friend修饰声名的类，为友元类。
在声明友元类时注意声名与定义的顺序问题--------先声明后定义。

如：class Boy
{
friend class Girl; //女孩类为男孩类的友元类
}
例如：下面的女孩类为男孩类的友元函数。

//友元类：在另外一个用friend修饰声名的类，成为友元类
class Boy
{
	friend class Girl;   //女孩类为男孩类的友元类
public:
	Boy(string name,int age);
	~Boy();
	
	void printBoy() {
		cout << "男孩中的输出函数：";
		cout << this->name<< "\t" <<this->age << endl;
	}
private:
	string name;
	int age;
};

Boy::Boy(string name, int age):name(name),age(age)
{
}

Boy::~Boy()
{
}

class Girl
{
public:
	Girl(string name,int age);
	~Girl();
	void print() {
		cout << "女孩中的输出函数：";
		cout << boy.name<< "\t" << boy.age << endl;
	}
private:
	Boy boy;          //所有boy里面的属性，对girl来说都是无视权限
};

Girl::Girl(string name, int age):boy(name,age)
{
}

Girl::~Girl()
{
}

注意：除了可以将一个类设置为另外一个类的友元类，还可以将这两个类互相设置为友元类。

二、特殊的运算符重载

  • = () -> [] 只能重载为类的成员函数
  • . .* ?: :: 不能被重载
  • 重载必须要存在一个自定义类型
  • 一个重载习惯: 单目运算类重载，双目采用友元重载
  • ++ – 重载
    • 前置和后置一个处理,后置通常会增加一个无用int参数作为标记
  • 输入输出重载
    • 流对象的类型
      • cout: ostream —>输出流对象
      • cin: istream -->输入流对象
    • 友元重载
    • 尽量保持用引用类型
  • 文本重载
    • 文本重载是后缀重载，不要写在类里面，也不要夹在几个类之间
  • 对象隐式转换
    • operator 要转化为的数据类型（）{ }
#include<iostream>
#include<thread>
#include<string>
using namespace std;

class MM
{
public:
	MM();
	MM(string name, int age);
	~MM();
	void printMM() {
		cout << "name: " << name << "\t" << "age: " << age << endl;
	}
	//i++    需用类重载
	MM operator++(int) {
		return MM(this->name, this->age++);          //注意：返回语句需加上返回类型，这里的返回类型是MM
	}
	//++i	 需用类重载
	MM operator--() {
		return MM(this->name, --this->age);       
		//或者：  --this.age;  return (this*)
	}

	//输入输出流重载
	friend istream & operator>> (istream& in, MM& object);
	friend ostream& operator<<(ostream& out, MM& object);

private:
	string name;
	int age;

};
MM::MM(string name, int age) :name(name), age(age) {}

istream& operator>> (istream& in, MM& object) {
	cout << "请依次输入名字和年龄：";
	in >> object.name >> object.age ;
	return in;
}
ostream& operator<<(ostream& out, MM& object) {
	out << object.name <<"\t" << object.age << endl;
	return out;
}


MM::MM()
{
}

MM::~MM()
{
}


//对象的隐式转换
class Boy
{
public:
	Boy(double height,int age);
	~Boy();
	//对象的隐式转换
	operator int() {
		return age;
	}

	int age;
	double height;
};

Boy::Boy(double height, int age):height(height),age(age)
{
}

Boy::~Boy()
{
}

//文本重载    即后缀重载          注意：文本重载不能成为类的成员，即不能写在类里面,最好也不要写在两个类之间
unsigned long long operator""_h(unsigned long long int hour) {
	return hour * 3600;
}

unsigned long long operator""_m(unsigned long long int min) {
	return min * 60;
}

unsigned long long operator""_s(unsigned long long int seconds) {
	return seconds;
}
