Linux - shell 脚本后台运行某条命令 - 输出为 pid 文件 - Shell脚本模板

Shell 模板脚本参数具体解释：

$! 输出命令运行后获取到的进程号，注意是 美元符号$和感叹号!

$? 的原理是shell使用关键字PID来获取最后一条命令及命令进程ID，
并用它来获取最后一条命令的执行状态，通过返回两个数字，
来判断执行结果：返回 0 表示命令执行成功，返回大于 0 表示命令失败。
例如：echo $? 
如果命令执行成功，则会输出 0

shell 脚本模板演示

测试文件名：execute.sh

启动命令：（如果不存在服务，则会启动服务，如果存在服务，则会自动重启服务）

sh execute.sh

sh execute

envstart：启动环境，如解释器等路径，替换为自己的解释器路径

file：替换为自己的要运行的文件路径

logfile：替换为自己的日志文件路径

pid_file：替换为自己的自己的要运行的文件夹路径/xxx.pid

#!/bin/bash
######################################
# author: Muxiu
# shell file: execute.sh
# shell command: sh execute
######################################
# 解释器路径
envstart=/bin/python
# 运行的文件路径
file=/home/test/test.py
# 日志文件路径
logfile=/home/test/mytest.log
# shell 脚本的当前路径
current_path=$(pwd)
# 配置 pid 文件名
pid_file=/home/test/test.pid
# 绝对路径配置 当前路径 斜杠 pid 文件名
output=$current_path/$pid_file
# 输出 pid 文件的绝对路径
echo $output
# 如果 pid 文件存在 -f
if [ -f "$output" ];then
    echo "pid文件存在"
    PID=$(cat $output)

    PID_EXIST=$(ps aux | awk '{print $2}'| grep -w $PID)
	# 如果 pid 文件存在，但是 pid 文件不存在 pid 进程号，找不到 pid 进程
    if [ ! $PID_EXIST ];then
	    echo the process $PID is not exist
	    echo starting $envstart $file command 新进程将会覆盖 pid 文件
		# 后台运行 —— 守护进程 
		# $! 输出命令运行后获取到的进程号，注意是 美元符号$和感叹号!
	    nohup $envstart $file > $logfile 2>&1 & echo $! > $output
	    echo the new process $! is running
	    echo $envstart $file command Started
	    echo Success Started
    else
		# 存在 pid 进程号
	    echo the process $PID exist
		# 删除进程，重新运行
	    kill -9 $PID
	    # 如果删除进程成功
	    if [ $? -eq 0 ]; then 
	    	echo "kill the process $PID" 
	    else
		    #删除进程失败
		    echo "failed kill the process $PID"
		    echo "$envstart $file command Stop"
		    return
	    fi
	    # 暂停 1 秒
	    echo need to sleep 1 s ...
	    sleep 1
	    echo 1 s ...
	
		# 后台运行 —— 守护进程
	    nohup $envstart $file > $logfile 2>&1 & echo $! > $output
	    echo the new process $! is running
	    echo $envstart $file command Started
	    echo Success restart
    fi
else
    echo "pid文件不存在"
    # 新建 pid 文件
    touch $output
    echo "新建 pid 文件："$output
	# 暂停 1 秒
    echo need to sleep 1 s ...
    sleep 1
    echo 1 s ...
	# 后台运行 —— 守护进程
	nohup $envstart $file > $logfile 2>&1 & echo $! > $output
    echo the new process $! is running
    echo $envstart $file command Started
    echo Success started

fi
