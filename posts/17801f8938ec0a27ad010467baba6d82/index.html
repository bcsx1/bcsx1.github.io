59、【图+树】最小生成树（BST）问题——Prim算法和Kruskal算法（C/C++版）

一、最小生成树

问题背景：

多个城市间铺设电路所花费的代价。

最小生成树是一个有 n 个结点的连通图的生成树是原图的极小连通子图，即n各点对应有n-1条边。同时，使其极小连通子图中边的权重之和最小。

在一个连通图中，含有n个顶点和n-1条边，可称该图为生成树，即连通子图。最小生成树就是在连通网里所有生成树中边权重之和最小的那个就是最小生成树。

最小生成树可以用kruskal（克鲁斯卡尔）算法或prim（普里姆）算法求出。

设图中节点数为n，边数为m。

当存储的图为稠密图时，常用朴素Prim算法O(n2)；当存储的图为稀疏图时，常用Kruskal算法O(m log2m)。

二、Prim算法

1、算法介绍

普里姆算法（Prim算法），图论中的一种算法，可在加权连通图里搜索最小生成树。意即由此算法搜索到的边子集所构成的树中，不但包括了连通图里的所有顶点（英语：Vertex (graph theory)），且其所有边的权值之和亦为最小。

2、题目描述

给定一个 n 个点 m条边的无向图，图中可能存在重边和自环，边权可能为负数。

求最小生成树的树边权重之和，如果最小生成树不存在则输出 impossible。

给定一张边带权的无向图 G=(V,E)，其中 V 表示图中点的集合，E 表示图中边的集合，n=|V|，m=|E|。

由 V中的全部 n 个顶点和 E 中 n−1 条边构成的无向连通子图被称为 G 的一棵生成树，其中边的权值之和最小的生成树被称为无向图 G的最小生成树。

输入格式

第一行包含两个整数 n和 m。
接下来 m行，每行包含三个整数 u,v,w，表示点 u 和点 v 之间存在一条权值为 w的边。

输出格式

共一行，若存在最小生成树，则输出一个整数，表示最小生成树的树边权重之和，如果最小生成树不存在则输出 impossible。

数据范围

1≤n≤500,
1≤m≤105,
图中涉及边的边权的绝对值均不超过 10000。

输入样例：

4 5
1 2 1
1 3 2
1 4 3
2 3 2
3 4 4

输出样例：

6

3、题目分析

算法实现过程与Dijkstra类似，可参考：

55、【图】Dijkstra求最短路径（单源最短路径+边权重为正数）（C/C++版）

区别在于Dijkstra每次求的是距离起始点最近的点，Prim每次加入的是距离集合最近的点。

每次新加入一个点后，由于已在dist数组中存储了未加入新点前其余点到集合的最近距离 ，因此需要将加入该点后集合距离其余各点距离进行更新。更新原则是 加入该点后通过新点到其余点的距离 与 不加入该点到其余点的距离 进行比较，选取距离最近的情况更新dist数组。

4、算法实现

#include <stdio.h>

const int N = 510, INF = 0x3f3f3f3f;        // 定义INF为无穷大
int n, m;
int g[N][N], dist[N];           // 邻接矩阵g、记录其余点到集合的最短距离数组dist
bool visit[N];                  // 标记节点i是否被访问过

int prim(){
    int bst = 0;        // 记录最小生成树的权重
    
    // 初始化dist
    for(int i = 0; i <= n; i++)     dist[i] = INF;
    dist[1] = 0;            // 选编号1的结点为起始点
    
    for(int i = 0; i < n; i++){
        int t = 0;
        for(int j = 1; j <= n; j++){        // 寻找距离集合最近的点加入到集合中
            if(!visit[j] && dist[j] < dist[t])      t = j;
        }
        if(t == 0)          return INF;     // 未遍历完n个点且也不能再找到距离集合更近的点，说明BST不存在
        visit[t] = true;                    // 标记加入的点已被访问过
        bst += dist[t];                     // 将集合至中转点的权重加入到BST权重之和中
         
        for(int j = 1; j <= n; j++){        // 更新加入新点后其余点到集合的最短距离
            if(g[t][j] < dist[j])         dist[j] = g[t][j];
        }
        
    }
    
    return bst;
}

int main(){
    scanf("%d%d", &n, &m);
    for(int i = 1; i <= n; i++)
        for(int j =1 ; j <=n; j++)
            g[i][j] = INF;
    
    while(m--){
        int x, y, z;       scanf("%d%d%d", &x, &y, &z);
        if(g[x][y] > z)    g[x][y] = g[y][x] = z;
    }

    int t = prim();
    if(INF == t)        puts("impossible");
    else                printf("%d", t);
    
    return 0;
}

#include <stdio.h>

const int N = 510, INF = 0x3f3f3f3f;
int g[N][N], dist[N];
bool visit[N];

int Prim(int n){
   for(int i = 0; i <= n; i++)      dist[i] = INF;
   dist[1] = 0;
   
   int res = 0;
   for(int i = 0; i < n; i++){
       int t = 0;
       for(int j = 1; j <= n; j++){
           if(!visit[j] && dist[j] < dist[t])
                t = j;
       }
       if(t == 0)       return INF;       
       visit[t] = true;
       res += dist[t];
       
       for(int j = 1; j <= n; j++){
           if(dist[j] > g[t][j])
                dist[j] = g[t][j];
       }
   }
   return res;
   
}

int main(){
    int n, m;       scanf("%d%d", &n, &m);
    for(int i = 0; i <= n; i++)
        for(int j = 0; j <= n; j++)
            if(i != j)  g[i][j] = INF;
            
    while(m--){
        int x, y, z;        scanf("%d%d%d", &x, &y, &z);
        if(g[x][y] > z)     g[x][y] = g[y][x] = z;
    }
    
    int res = Prim(n);
    if(res == INF)      puts("impossible");
    else                printf("%d", res);
    
    return 0;
}

时间复杂度是 O(n2+m), n 表示点数，m 表示边数

三、Kruskal算法

1、算法介绍

克鲁斯卡尔算法是求连通网的最小生成树的另一种方法。与普里姆算法不同，它的时间复杂度为O（eloge）（e为网中的边数），所以，适合于求边稀疏的网的最小生成树。

2、问题描述

给定一个 n 个点 m条边的无向图，图中可能存在重边和自环，边权可能为负数。

求最小生成树的树边权重之和，如果最小生成树不存在则输出impossible。

给定一张边带权的无向图 G=(V,E)，其中 V 表示图中点的集合，E 表示图中边的集合，n=|V|，m=|E|。

的最小生成树。

输入格式

第一行包含两个整数 n和 m。
接下来 m行，每行包含三个整数 u,v,w，表示点 u 和点 v 之间存在一条权值为 w的边。

输出格式

3、题目分析

Kruskal算法步骤：
Step 1： 将图中的边按权重由小到大进行排序。
Step 2： 按权重大小从小到大依次选边，并保证加入的边不会使已选取的集合中出现环路。
Step 3： 将所有边选择完后，判断是否构成了最小生成树（特征：树中n的点对应n - 1条边）。

4、算法实现

[ 前置知识 ]

（1）并查集

具体实现上，使用并查实来确保每次加入的边不会使其出现环路。

46、【树和二叉树】并查集——合并集合 （C/C++版）

每次选取一个边时，就将构成边的点加入到已选取的集合中。在下次选取边时，判断这个边对应的两个顶点是否在同一集合中。若在同一集合中，则不选取该边，查找下一个边。若不在同一集合中，则将另一个点加入已选取点的集合中，实现选取该边的效果。

（2）快速排序

使用快速排序将边按权重大小，由小到大进行排序。

36、【排序】快速排序实现（C/C++版）

#include <stdio.h>

const int N = 1e5 + 10, M = 2*1e5 + 10, INF = 0x3f3f3f3f;       // INF标记无最小生成树
int p[N];       // 并查集：P[x]为x的父节点
typedef struct Edge{        // 两点和一边的结构体
    int a, b, w;
}Edges[M];

int find(int x){        // 并查集
    if(p[x] != x)        p[x] = find(p[x]);
    return p[x];
}

void quick_sort(Edge *q, int l, int r){     // 使用快排实现对边排序
    if(l >= r)      return ;

    int x = q[l + r >> 1].w;
    int i = l - 1, j = r + 1;
    while(i < j){
        do i++;     while(q[i].w < x);
        do j--;     while(q[j].w > x);
        // 有一个非常重要的重点！！！ 交换的时候一定要交换整个结构体而不是结构体元素，这个细节一定不要漏了！
        // 因为这个细节一直Wrong了好久。。。
        if(i < j){
            Edge tmp = q[i];
            q[i] = q[j];
            q[j] = tmp;
        }
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int kruskal(Edge *edges, int n, int m){
    quick_sort(edges, 0, m - 1);                // 对边进行排序
    for(int i = 1; i <= n; i++)     p[i] = i;   // 初始化Trie树

    int res = 0, cnt = 0;                       // 存储结果和加入边的个数
    for(int i = 0; i < m; i++){                 // 按权值大小从小到大遍历
        int a = edges[i].a, b = edges[i].b, w = edges[i].w;
        if(find(a) != find(b)){                 // 点a和点b不在同一集合中，则将该一点加入集合中
            p[find(a)] = p[b];                  // 将a的根节点指向b
            res += w;
            cnt++;
        }   
    }
    return cnt == n - 1 ? res : INF;
}

int main(){
    int n, m;       scanf("%d%d", &n, &m);
    Edges edges;

    for(int i = 0; i < m; i++){
        int a, b, w;        scanf("%d%d%d", &a, &b, &w);
        edges[i] = { a, b, w };
    }

    int t = kruskal(edges, n, m);
    if(t == INF)
        puts("impossible");
    else
        printf("%d", t);

    return 0;
}

#include <stdio.h>

const int N = 1e5 + 10, M = 2*1e5 + 10, INF = 0x3f3f3f3f;
int p[N];
typedef struct Edge{
    int a, b, w;
}Edges[M];

int find(int x){
    if(p[x] != x)       p[x] = find(p[x]);
    return p[x];
}

void quick_sort(Edge *q, int l, int r){
    if(l >= r)      return ;

    int x = q[l + r >> 1].w;
    int i = l - 1, j = r + 1;
    while(i < j){
        do i++;     while(q[i].w < x);
        do j--;     while(x < q[j].w);
        if(i < j){
            Edge tmp = q[i];
            q[i] = q[j];
            q[j] = tmp;
        }
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int kruskal(Edge *edges, int n, int m){
    quick_sort(edges, 0, m - 1);
    for(int i = 1; i <= n; i++)     p[i] = i;

    int res = 0, cnt = 0;
    for(int i = 0; i < m; i++){
        int a = edges[i].a, b = edges[i].b, w = edges[i].w;
        if(find(a) != find(b)){
            p[find(a)] = p[b];
            res += w;
            cnt++;
        }
    }
    return cnt == n - 1 ? res : INF;
}

int main(){
    int n, m;       scanf("%d%d", &n, &m);
    Edges edges;

    for(int i = 0; i < m; i++){
        int a, b, w;        scanf("%d%d%d", &a, &b, &w);
        edges[i] = { a, b, w };
    }

    int t = kruskal(edges, n, m);
    if(t == INF)
        puts("impossible");
    else
        printf("%d", t);

    return 0;
}
