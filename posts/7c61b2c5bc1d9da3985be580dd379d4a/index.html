用c++实现特殊约瑟夫问题

题目描述:

编号为1…N的N个小朋友玩游戏，他们按编号顺时针围成一圈，从第一个人开始按逆时针次序报数，报到第M个人出列；然后再从下个人开始按顺时针次序报数，报到第 K 个人出列；再从下一个人开始按逆时针次序报数，报到第 M 个人出列；再从下个人开始按顺时针次序报数，报到第K个人出列……以此类推不断循环，直至最后一人出列。请编写程序按顺序输出出列人的编号。

输入格式:

输入为3个正整数，分别表示 N、M、K，均不超过1000

输出格式:

输出为一行整数，为出列人的编号。每个整数后一个空格。

输入样例:

6 3 5

输出样例:

5 3 1 2 4 6

根据题目逐句分析问题要如何实现

编号为1…N的N个小朋友玩游戏，他们按编号顺时针围成一圈

这个很简单，我们创建一个数组（或者链表），用for循环逐个编号即可

要注意的是由于题目要求编号围成一圈，我们要使用圆圈数组

好吧，这具体叫什么数组我也不知道，或许叫循环数组？总之就是将数组的头部和尾部连接起来，具体要怎么实现呢？

当上溢时，我们用%符号取余实现，当下溢出时，我们讲地址加上数组的长度实现

if (ptr >= lenth)
		return ptr % lenth;//上溢解决
	else if (ptr < 0)
		return ptr + lenth;//下溢解决
	//ptr为当前操作的数组地址，lenth为数组长度

我们写出顺时针报数和逆时针报数时更新操作指针ptr的代码

int NiCircle(int ptr,int lenth)
{
	//对指针值进行修改使得其不断逆时针循环
	ptr--;
	if (ptr >= lenth)
		return ptr % lenth;//上溢解决
	else if (ptr < 0)
		return ptr + lenth;//下溢解决
	else return ptr;
}
int ShunCircle(int ptr, int lenth)
{
	//对指针值进行修改使得其不断顺时针循环
	ptr++;
	if (ptr >= lenth)
		return ptr % lenth;//上溢解决
	else if (ptr < 0)
		return ptr + lenth;//下溢解决
	else return ptr;
}

从第一个人开始按逆时针次序报数，报到第M个人出列；

这个算法的实现也很简单，开始前设置一个计数器count并初始化为0，操作地址ptr初始化为0指向数组的第一个元素，开始报数，先检测操作地址ptr指向的数组元素是否为-1（用-1来标识这个位置的小朋友是否已经出列），不是则计数器count+1，ptr-1（逆时针），一直到count=M，即已经有M个小朋友报数，我们讲该小朋友的编号输出，并将该位置的编号置为-1即可

//逆时针出列
			int count = 0;
		int temp = 0;
		while (count != M)
		{
			if (Arr[ptr] != -1)
			{
				count++;
			}
			temp = ptr;
			ptr =NiCircle(ptr, N);
		}
		//出while循环表示已经逆时针找到了M，1.指向点输出 2.指向点置为-1 3.lenth-1 4.更新ptr 找到下一个人
		cout << Arr[temp] << " ";
Arr[temp] = -1;

值得注意的是，这里的ptr指向的地址已经是第M个小朋友的下一个小朋友了，为了输出方便，我们需要加一个辅助指针temp来保存ptr指针的上一次指向的地方

	int temp1 = temp;
		lenth--;
		temp = NiCircle(temp, N);
		while (Arr[temp] == -1)
		{
			temp = NiCircle(temp, N);
		}
		Arr[temp1] = -1;
		ptr = temp;
//找到下个人的位置

然后再从下个人开始按顺时针次序报数，报到第 K 个人出列；

逆时针报数ptr是每次-1，而顺时针报数ptr是每次+1，其余操作是完全一样的，我们Ctrl+c Ctrl即可

//顺时针出列
			int count = 0;
		int temp = 0;
		while (count != K)
		{
			if (Arr[ptr] != -1)
			{
				count++;
			}
			temp = ptr;
			ptr = ShunCircle(ptr, N);
		}
		//出while循环表示已经逆时针找到了M，1.指向点输出 2.指向点置为-1 3.lenth-1 4.更新ptr 找到下一个人
		cout << Arr[temp] << " ";
		int temp1 = temp;
		lenth--;
		temp = ShunCircle(temp, N);
		while (Arr[temp] == -1)
		{
			temp = ShunCircle(temp, N);
		}
		Arr[temp1] = -1;
		ptr = temp;

再从下一个人开始按逆时针次序报数，报到第 M 个人出列；再从下个人开始按顺时针次序报数，报到第K个人出列……以此类推不断循环，直至最后一人出列。

在之前的基础上，我们加上一个判断条件，使得第一次用逆时针报数，第二次用顺时针报数....一直到数组中的编号全部输出为之，算法就结束了

完整代码如下

#include<iostream>
using namespace std;
int NiCircle(int ptr,int lenth)
{
	//对指针值进行修改使得其不断逆时针循环
	ptr--;
	if (ptr >= lenth)
		return ptr % lenth;//上溢解决
	else if (ptr < 0)
		return ptr + lenth;//下溢解决
	else return ptr;
}
int ShunCircle(int ptr, int lenth)
{
	//对指针值进行修改使得其不断顺时针循环
	ptr++;
	if (ptr >= lenth)
		return ptr % lenth;//上溢解决
	else if (ptr < 0)
		return ptr + lenth;//下溢解决
	else return ptr;
}
int main()
{
	int N,M,K;
	cin >> N;
	cin >> M;
	cin >> K;
	//输入人数
	int Arr[1000];
	for (int i = 0; i < N; i++)
		Arr[i] = i + 1;
	//编号
	int ptr = 0;//循环数组的指针
	int lenth = N;//数组的长度
	int Judge = 2;//判断是顺时针还是逆时针的Judge
	
	
	while (lenth != 0)
	{
		if(Judge%2==0)
		{
			//逆时针出列
			int count = 0;
		int temp = 0;
		while (count != M)
		{
			if (Arr[ptr] != -1)
			{
				count++;
			}
			temp = ptr;
			ptr =NiCircle(ptr, N);
		}
		//出while循环表示已经逆时针找到了M，1.指向点输出 2.指向点置为-1 3.lenth-1 4.更新ptr 找到下一个人
		cout << Arr[temp] << " ";
		int temp1 = temp;
		lenth--;
		temp = NiCircle(temp, N);
		while (Arr[temp] == -1)
		{
			temp = NiCircle(temp, N);
		}
		Arr[temp1] = -1;
		ptr = temp;
		}
		else
		{
			//逆时针出列
			int count = 0;
		int temp = 0;
		while (count != K)
		{
			if (Arr[ptr] != -1)
			{
				count++;
			}
			temp = ptr;
			ptr = ShunCircle(ptr, N);
		}
		//出while循环表示已经逆时针找到了M，1.指向点输出 2.指向点置为-1 3.lenth-1 4.更新ptr 找到下一个人
		cout << Arr[temp] << " ";
		int temp1 = temp;
		lenth--;
		temp = ShunCircle(temp, N);
		while (Arr[temp] == -1)
		{
			temp = ShunCircle(temp, N);
		}
		Arr[temp1] = -1;
		ptr = temp;
			
		}
		Judge++;
	}
	return 0;
}
