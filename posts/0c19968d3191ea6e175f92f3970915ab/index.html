#include<stdio.h>
#define N 10
int main()
{
	float score[N],s=0;
	int i;
	for(i=1;i<N;i++)
	{
		scanf("%f",&score[i]);
		s=s+score[i];
	}
	printf("avervage=%.lf\n",s/N);
return 0;}

 6.判断回文

#include <stdio.h>
int main()
{
	char s[100];
	int i=0,j,n,flag=0;
	while ((s[i]=getchar())!='\n')
		i++;
	if(n=(i%2==0))
		n=i/2;
		else n=i/2+1;
	for (j=0;j<n;j++)
	{
		if (s[j]!=s[i-j-1])
		{
			flag=1;
			break;	}	
	}
	if (flag==0)
		printf("yes");
	else
		printf("no");
	return 0;}
