搜索二叉树

基础概念

学习原因:
map和set特性需要先铺垫二叉搜索树，而二叉搜索树也是一种树形结构
性质:

  1. 若它的左子树不为空，则左子树上所有节点的值都小于根节点的值
  2. 若它的右子树不为空，则右子树上所有节点的值都大于根节点的值
  3. 它的左右子树也分别为二叉搜索树

功能:

  1. 可以搜索
  2. 自带排序功能
  3. 去重,重复值插入失败

搜索二叉树的结构由插入顺序决定

二叉搜索树的性能分析

插入和删除操作都必须先查找，查找效率代表了二叉搜索树中各个操作的性能。
对有n个结点的二叉搜索树，若每个元素查找的概率相等，则二叉搜索树平均查找长度是结点在二叉搜索树的深度的函数，即结点越深，则比较次数越多。
但对于同一个关键码集合，如果各关键码插入的次序不同，可能得到不同结构的二叉搜索树：


最优情况下，二叉搜索树为完全二叉树(或者接近完全二叉树)，其平均比较次数为：log_2(N)
最差情况下，二叉搜索树退化为单支树(或者类似单支)，其平均比较次数为：N

#pragma once
#include <iostream>
using namespace std;

//二叉搜索树	数据结构
namespace Key
{
	template<class K>
	struct BSTreeNode
	{
		BSTreeNode<K>* _left;
		BSTreeNode<K>* _right;
		K _key;

		BSTreeNode(const K& key)
			:_left(nullptr)
			, _right(nullptr)
			, _key(key)
		{}
	};

	template<class K>
	class BSTree
	{
		typedef BSTreeNode<K> Node;
	public:
		// 默认构造, 显示写了构造(拷贝构造),编译器就不生成默认构造,可以用default强制生成
		BSTree() = default; 

		BSTree(const BSTree<K>& b)
		{
			CopyTree(_root, b._root);
		}

		~BSTree()
		{
			//cout << "~BSTree()" << endl;
			DestoryTree(_root);
		}

		BSTree<K>& operator=(BSTree<K> b)
		{
			//拷贝后的b,不用判断if(&b != this);本就不同
			//this->_root换到临时对象b,调用结束,被析构了
			std::swap(_root, b._root);
			return *this;
		}

		bool Insert(const K& key)
		{
			if (_root == nullptr)
			{
				_root = new Node(key);
				return true;
			}
			else
			{
				//找到插入位置
				Node* parent = _root;
				Node* cur = _root;
				while (cur)
				{
					if (cur->_key > key)
					{
						parent = cur;
						cur = cur->_left;
					}
					else if (cur->_key < key)
					{
						parent = cur;
						cur = cur->_right;
					}
					else
					{
						//相等,插入失败
						return false;
					}
				}

				cur = new Node(key);
				if (parent->_key > key)
				{
					parent->_left = cur;
				}
				else
				{
					parent->_right = cur;
				}
				return true;
			}
		}

		bool Find(const K& key)
		{
			if (_root == nullptr)
			{
				return false;
			}

			Node* cur = _root;
			while (cur)
			{
				if (cur->_key > key)
				{
					cur = cur->_left;
				}
				else if (cur->_key < key)
				{
					cur = cur->_right;
				}
				else
				{
					return true;
				}
			}

			return false;
		}

		bool Erase(const K& key)
		{
			if (_root == nullptr)
			{
				cout << "empty!" << endl;
				return false;
			}

			Node* cur = _root;
			Node* parent = _root;
			while (cur)
			{
				//找到key
				if (cur->_key > key)
				{
					parent = cur;
					cur = cur->_left;
				}
				else if (cur->_key < key)
				{
					parent = cur;
					cur = cur->_right;
				}
				else
				{
					//debug: 需要排除cur是root情况
					//cur左右子树其中一个为空,父节点继承单个就行
					if (cur->_left == nullptr)
					{
						if (cur == _root)
						{
							_root = cur->_right;
						}
						else
						{
							if (parent->_key > cur->_key)
							{
								parent->_left = cur->_right;
							}
							else
							{
								parent->_right = cur->_right;
							}
						}

						delete cur;
					}
					else if (cur->_right == nullptr)
					{
						if (cur == _root)
						{
							_root = cur->_left;
						}
						else
						{
							if (parent->_key > cur->_key)
							{
								parent->_left = cur->_left;
							}
							else
							{
								parent->_right = cur->_left;
							}
						}

						delete cur;
					}
					else
					{
						// 找到了_key,开始删除
						// 1.cur左右子树其中一个为空,父节点继承单个就行
						// 2.cur左右子树都不为空,找一个可以替代父节点的监护人赋值父节点,然后删除监护人节点
						// 替代父节点的监护人: 左子树的最大值 or 右子树的最小值 -- 问题转化为1.

						// 1.cur左右子树其中一个为空
						// debug: 需要排除cur是root情况,更换root节点
						Node* pmaxLeft = cur;
						Node* maxLeft = cur->_left;
						while (maxLeft->_right)
						{
							pmaxLeft = maxLeft;
							maxLeft = maxLeft->_right;
						}
						cur->_key = maxLeft->_key;

						//maxLeft可能有左子树
						// 2.0
						if (pmaxLeft->_left == maxLeft)	//cur == pmaxLeft
						{
							pmaxLeft->_left = maxLeft->_left;
						}
						else
						{
							pmaxLeft->_right = maxLeft->_left;
						}
						delete maxLeft;
					}
					return true;
				}
			}

			return false;
		}

		void InOrder()
		{
			//需要this指针才能找到成员,但是this指针也做为函数参数
			//因为this指针原因,成员遍历不能作为缺省参数 -> 子函数解决递归传参
			_InOrder(_root);
			cout << endl;
		}

		//递归版本
		bool InsertR(const K& key)
		{
			return _InsertR(_root, key);
		}

		bool EraseR(const K& key)
		{
			return _EraseR(_root, key);
		}

		bool FindR(const K& key)
		{
			return _FindR(_root, key);
		}

	protected:
		// 注意引用传参
		void CopyTree(Node*& copy, Node* cur)
		{
			if (cur == nullptr)
				return;

			copy = new Node(cur->_key);
			CopyTree(copy->_left, cur->_left);
			CopyTree(copy->_right, cur->_right);
		}

		void DestoryTree(Node* cur)
		{
			if (cur == nullptr)
				return;
			//后序遍历
			DestoryTree(cur->_left);
			DestoryTree(cur->_right);
			//cout << cur->_key << endl;
			delete cur;
		}

		bool _EraseR(Node*& root, const K& key)
		{
			if (root == nullptr)
			{
				return false;
			}

			if (root->_key > key)
			{
				return _EraseR(root->_left, key);
			}
			else if (root->_key < key)
			{
				return _EraseR(root->_right, key);
			}
			else
			{
				//找到了,开始删除,root
				//单侧为空,引用的妙用
				Node* del = root;
				if (root->_left == nullptr)
				{
					root = root->_right;
				}
				else if (root->_right == nullptr)
				{
					root = root->_left;
				}
				else
				{
					//两侧都不为空, 找左子树最大 or 右子树最小
					Node* maxLeft = root->_left;
					while (maxLeft->_right)
					{
						maxLeft = maxLeft->_right;
					}

					swap(root->_key, maxLeft->_key);
					_EraseR(root->_left, key);	//交换后root->_key < key; 只能去左子树删除
				}
				delete del;
			}
		}

		bool _InsertR(Node*& root, const K& key)
		{
			if (root == nullptr)
			{
				//root为父节点_left 或 _right的引用,可直接赋值
				root = new Node(key);
				return true;
			}

			if (root->_key > key)
			{
				return _InsertR(root->_left, key);
			}
			else if (root->_key < key)
			{
				return _InsertR(root->_right, key);
			}
			else
			{
				return false;
			}
		}

		bool _FindR(Node* root, const K& key)
		{
			if (root == nullptr)
				return false;

			if (root->_key > key)
			{
				return _FindR(root->_left, key);
			}
			else if (root->_key < key)
			{
				return _FindR(root->_right, key);
			}
			else
			{
				return true;
			}
		}

		//中序遍历
		void _InOrder(Node* root)
		{
			if (root == nullptr)
			{
				return;
			}

			_InOrder(root->_left);
			cout << root->_key << " ";
			_InOrder(root->_right);
		}

	private:
		Node* _root = nullptr;
	};
}

主要功能

1.二叉搜索树的查找

a、从根开始比较，查找，比根大则往右边走查找，比根小则往左边走查找。
b、最多查找高度次，走到到空，还没找到，这个值不存在。
这里不用返回指向该节点的指针,因为修改节点_key,可能会破坏树的结构.

bool Find(const K& key)

		{
			cur = cur->_right;
		}
		else
		{
			return true;
		}
	}

	return false;
}
2.二叉搜索树的插入

插入的具体过程如下：
a. 树为空，则直接新增节点，赋值给_root指针
b. 树不空，按二叉搜索树性质查找插入位置，插入新节点
c. 如果插入的值已经存在, 则插入失败

				cur = cur->_right;
			}

		}

		}

	}
}
3.二叉搜索树的删除

首先查找元素是否在二叉搜索树中，如果不存在，则返回, 否则要删除的结点可能分下面四种情况：
a. 要删除的结点无孩子结点
b. 要删除的结点只有左孩子结点
c. 要删除的结点只有右孩子结点
d. 要删除的结点有左、右孩子结点

情况a：直接删除,可以归类到情况b,c
情况b：删除该结点且使被删除节点的双亲结点指向被删除节点的左孩子结点--直接删除
情况c：删除该结点且使被删除节点的双亲结点指向被删除结点的右孩子结点--直接删除
情况d：在它的右子树中寻找中序下的第一个结点(关键码最小)，用它的值填补到被删除节点中，再来处理该结点的删除问题--替换法删除

找到了_key,开始删除
1.cur左右子树其中一个为空,父节点继承单个就行
2.cur左右子树都不为空,找一个可以替代父节点的"监护人"赋值父节点,然后删除监护人节点.
替代父节点的监护人: 左子树的最大值 or 右子树的最小值 -- 问题转化为1(cur左右子树其中一个为空)


情况b：删除该结点且使被删除节点的双亲结点指向被删除节点的左孩子结点--直接删除
情况c：删除该结点且使被删除节点的双亲结点指向被删除结点的右孩子结点--直接删除


情况d：在它的右子树中寻找中序下的第一个结点(关键码最小)，用它的值填补到被删除节点中，再来处理该结点的删除问题--替换法删除
左右子树都不为空,选介于左右子树大小之间的值作为cur子树根节点
"可以替代的节点" --  左子树的最大值 or 右子树的最小值
左子树 < 父节点 < 右子树
左子树 < 左子树最大值 or 右子树最小值 < 右子树



注意:需要排除cur是root情况,解决:更换root节点

					else

5.通过递归实现查找,插入,删除

当原函数不方便递归的时候,可以利用子函数递归;
可以画递归展开图辅助理解递归;

这里bool _EraseR(Node*& root, const K& key)
需要留意,第一个参数类型Node*&, 引用传参
当递归到可以删除的位置时,root为它的父节点parent->_left or parent->_right的引用
修改root就是修改父节点parent左右指针的指向
那么就不用判断root在其父节点parent的那一侧,直接赋值

eg:
if (root->_left == nullptr)
{
	root = root->_right;  ==>等价于 parent->指向root一侧的指针 = root->_right;
}

bool _EraseR(Node*& root, const K& key)
{
	if (root == nullptr)
	{
		return false;
	}

	if (root->_key > key)
	{
		return _EraseR(root->_left, key);
	}
	else if (root->_key < key)
	{
		return _EraseR(root->_right, key);
	}
	else
	{
		//找到了,开始删除,root
		//单侧为空,引用的妙用
		Node* del = root;
		if (root->_left == nullptr)
		{
			root = root->_right;
		}
		else if (root->_right == nullptr)
		{
			root = root->_left;
		}
		else

			swap(root->_key, maxLeft->_key);
			_EraseR(root->_left, key);	//交换后root->_key < key; 只能去左子树删除
		}
		delete del;
	}
}

		//root为父节点_left 或 _right的引用,可直接赋值

		return true;
	}

		return _InsertR(root->_left, key);
	}
	else if (root->_key < key)
	{
		return _InsertR(root->_right, key);
	}
	else

	}
}

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        if(nullptr == root)
        {
            return vector<vector<int>>();
        }

        //利用队列和一个记录每次个数的变量
        //队列中最多同时存在两层数据,上一层pop完,队列剩下的就是下一层
        queue<TreeNode*> q;
        q.push(root);
        int levelSize = 1;
        vector<vector<int>> vv;
        while(!q.empty())
        {
            vector<int> v;
            while(levelSize--)
            {
                TreeNode* front = q.front();
                q.pop();
                v.push_back(front->val);

                if(front->left)
                    q.push(front->left);
                if(front->right)
                    q.push(front->right);
            }

            levelSize = q.size();
            vv.push_back(v);
        }

        return vv;
    }
};

方法二:

2.0 代码优化  时间复杂度O(N^2):最会情况,全歪脖子树
思路:
1.如果根是p或q,那么就是公共祖先
2.如果p和q分别在一个节点左子树和右子树,那么这个节点就是公共祖先
3.如果p和q同时在左子树/右子树,那么就到左子树/右子树去寻找:转为为子问题

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        //2.0 代码优化  时间复杂度O(N^2):最会情况,全歪脖子树
        //思路:
        //1.如果根是p或q,那么就是公共祖先
        //2.如果p和q分别在一个节点左子树和右子树,那么这个节点就是公共祖先
        //3.如果p和q同时在左子树/右子树,那么就到左子树/右子树去寻找:转为为子问题
        if(root == p || root == q)
            return root;

        bool pinLeft = isInTree(root->left, p);
        bool pinRight = !pinLeft;
        bool qinLeft = isInTree(root->left, q);
        bool qinRight = !qinLeft;
        if((pinLeft && qinRight)
        || (pinRight && qinLeft))
        {
            return root;
        }

        if(pinLeft && qinLeft)
            return lowestCommonAncestor(root->left, p, q);
        if(pinRight && qinRight)
            return lowestCommonAncestor(root->right, p, q);

        return root;
    }

    bool isInTree(TreeNode* root, TreeNode* x)
    {
        if(root == nullptr)
            return false;

        if(root == x)
            return true;
        else
            return isInTree(root->left, x) || isInTree(root->right, x);
    }
};

class Solution {

    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        //3.0 代码优化  时间复杂度O(N)
        //思路:
        //1.利用栈记录p,q路径
        //2.栈的元素就是每层经过的节点
        //3.公共祖先到根节点的距离一样,在两个栈中栈底到公共节点距离一样
        //4.pop长的栈,使两个栈长度,栈顶相同则是公共祖先,否则pop,继续比较
        stack<TreeNode*> pPath;
        stack<TreeNode*> qPath;
        getPath(root, p, pPath);
        getPath(root, q, qPath);

        while(pPath.size() != qPath.size())
        {
            if(pPath.size() > qPath.size())
                pPath.pop();
            else
                qPath.pop();
        }

        while(pPath.top() != qPath.top())
        {
            pPath.pop();
            qPath.pop();
        }
        return pPath.top();
    }

            return true;
        }
        else
        {   //关键条件
            st.pop();
            return false;
        }  
    }

                return;

            //引用,解决初始左右节点为空问题
            TreeNode* node  = new TreeNode(preorder[0]);
            root = node;
            return;
        }

        //子树根节点处理
        int rootVal = preorder[0];
        TreeNode* _root = new TreeNode(rootVal);

        //拆分区间,左子树元素 根 右子树元素
        int left_size = 1;
        vector<int> in_left,in_right,pre_left,pre_right;
        if(inorder.size() > 1)
        {
            //匿名对象,利用迭代器区间初始化构造
            auto in_left_end = find(inorder.begin() ,inorder.end(), rootVal);
            in_left = vector<int>(inorder.begin(), in_left_end);
            in_right = vector<int>(++in_left_end, inorder.end());

            int left_size = in_left.size();
            pre_left = vector<int>(++preorder.begin(), ++preorder.begin() + left_size);
            pre_right = vector<int>(++preorder.begin() + left_size, preorder.end());
        }

        _buildTree(pre_left, in_left, _root->left);
        _buildTree(pre_right, in_right, _root->right);
        root = _root;
        return;
    }

};

//2.0 代码优化  空间复杂度 O(1)
//思路:
//1.利用根节点拆分区间,就像快速排序 区间: 左子树区间 根 右子树区间
//2.利用数组下标记录拆分后的区间,避免开辟空间
//当区间元素等于1,new一个节点,链接; 区间不存在,则nullptr

class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        //2.0 代码优化  空间复杂度 O(1)
        //思路:
        //1.利用根节点拆分区间,就像快速排序 区间: 左子树区间 根 右子树区间
        //2.利用数组下标记录拆分后的区间,避免开辟空间
        //当区间元素等于1,new一个节点,链接; 区间不存在,则nullptr
        //子区间根节点下标,rooti从左子树依次指向右子树根节点下标
        int rooti = 0;
        int begin = 0; 
        int end = inorder.size() - 1;
        TreeNode* root = _buildTree(preorder, inorder, begin, end, rooti);
        return root;
    }

    TreeNode* _buildTree(vector<int>& preorder, vector<int>& inorder, int begin, int end, int& rooti) 
    {	//当区间元素等于1,new一个节点,链接; 区间不存在,则nullptr
        if(begin >= end)
        {
            if(begin == end)
            {
                TreeNode* node = new TreeNode(preorder[rooti++]);
                return node;
            }
            else
            {
                return nullptr;
            }
        }

        int div = begin;
        while(inorder[div] != preorder[rooti])
        {
            div++;
        }

        TreeNode* root = new TreeNode(preorder[rooti++]);
        //pre前序, 左子树 根 右子树 
        //顺序,rooti先指向左子树根节点
        root->left = _buildTree(preorder, inorder, begin, div - 1, rooti);
        root->right = _buildTree(preorder, inorder, div + 1, end, rooti);
        return root;
    }
};

class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        //2.0
        //思路:
        //1.非递归, 利用数据结构栈LIFO
        //2.将问题转化为左路节点和它的右子树
        //3.前序,先遍历左路节点,并且入栈
        //4.左为nullptr,遍历它的右子树 : 子问题
        //5.左右遍历结束,pop栈顶
        if(root == nullptr)
            return vector<int>();

        stack<TreeNode*> st;
        vector<int> v;
        //右路节点不为空/栈不为空
        while(root || !st.empty())
        {
            //1.先遍历左路节点,并且入栈
            while(root)
            {
                v.push_back(root->val);
                st.push(root);
                root = root->left;
            }
            //2.左路节点的右子树 : 子问题
            TreeNode* top = st.top();
            st.pop();
            //3.子问题访问右子树
            root = top->right;
        }

        return v;
    }
};

class Solution {

    vector<int> postorderTraversal(TreeNode* root) {
        //非递归
        if(nullptr == root)
            return vector<int>();

        stack<int> stVal;
        stack<TreeNode*> stNode;
        vector<int> v;
        stNode.push(root);
        //后序,只入数据,中途不出数据
        while(!stNode.empty())
        {
            TreeNode* top = stNode.top();
            stNode.pop();
            stVal.push(top->val);

            //顺序变了
            if(top->left)
                stNode.push(top->left);
            if(top->right)
                stNode.push(top->right);
        }
        //最后再全出数据
        while(!stVal.empty())
        {
            v.push_back(stVal.top());
            stVal.pop();
        }
        return v;
    }
};

class Solution {
public:

        //思路:
        //1.非递归, 利用数据结构栈LIFO
        //2.左路节点和它的右子树
        //3.用一个prev保存上一个pop节点,判断是否是根的右子树
        //4.不是右子树,子问题访问右子树,是,则访问过,可以pop

        TreeNode* prev = nullptr;
        //第一次根节点不为空,进入循环
        while(root || !st.empty())
        {
            //访问过的节点不再入栈
            if(root != prev)

                while(root)

                }

            //2.右子树不为空,但访问过了,可以访问根节点了
            if(top->right == nullptr || top->right == prev)

                {

            }
            //3.右子树不为空,且没访问过,先访问右子树
            root = top->right;
        }

        return v;
    }
