<template>

        </template>
      </van-button>
    </div>
  </div>

<script>
import _config from '@/config/server';
import { merchantReceipt } from '@/request/api/hospital/subscribeOrders';
export default {
  data() {
    return {
      out_trade_no: '',
      jumpOutUrl: '',
      ticketInfo: {},
    };
  },
  created() {
    /** 点金小票的js，需要在head中引入 */
    let head = document.querySelector('head');
    let script = document.createElement('script');
    script.type = 'text/javascript';
    script.src = 'https://wx.gtimg.com/pay_h5/goldplan/js/jgoldplan-1.0.0.js';
    head.appendChild(script);

    // 解构微信传过来的参数字段
    const { out_trade_no, sub_mch_id, check_code } = this.$route.query;
    const url = 'https://yishengh5.sr-2.com/#/paySuccess/index';
    console.log('url', url);
    console.log('out_trade_no', out_trade_no);
    console.log('sub_mch_id', sub_mch_id);
    console.log('check_code', check_code);
    // 判断如果字段都有值
    if (out_trade_no && sub_mch_id && check_code) {
      merchantReceipt({
        url: url,
        sub_mch_id: sub_mch_id,
        out_trade_no: out_trade_no,
        check_code: check_code,
      }).then((res) => {
        if (res.code == 200) {
          // 获取当前环境url前缀
          let url = JSON.parse(JSON.stringify(_config.apiUrl));
          url = url.replace('prod-api', '#').replace('api', '#');
          let num = [10, 30];
          if (num.includes(res.data.subscribeType)) {
            // 远程和预约
            this.jumpOutUrl = url + 'appointmentsuccess';
          } else if (res.data.subscribeType == 20) {
            //体检套餐
            this.jumpOutUrl = url + 'successPage';
          }
          console.log('this.jumpOutUrl', this.jumpOutUrl);
          this.ticketInfo = {
            orderSn: res.data.orderSn,
            payTime: res.data.payTime,
            orderAmountStr: res.data.orderAmountStr,
          };
        }
      });
      // this.out_trade_no = out_trade_no;
      // 发送请求,调用后端接口获取订单信息并展示
      // this.getOrdersInfo(this.out_trade_no);
    }
  },

  mounted() {
    this.wxInit();
  },
  methods: {
    //注册点击事件(去首页)
    JumpToHome() {
      var mchData = {
        action: 'jumpOut',
        jumpOutUrl: this.jumpOutUrl, //跳转的页面
      };
      const postData = JSON.stringify(mchData);
      window.parent.postMessage(postData, 'https://payapp.weixin.qq.com');
    },
    // 微信商家小票
    wxInit() {
      //初始化小票
      let mchData = {
        action: 'onIframeReady',
        displayStyle: 'SHOW_CUSTOM_PAGE',
        //   displayStyle: this.out_trade_no
        //     ? 'SHOW_CUSTOM_PAGE'
        //     : 'SHOW_OFFICIAL_PAGE',
      };
      let postData = JSON.stringify(mchData);
      window.parent.postMessage(postData, 'https://payapp.weixin.qq.com');
    },
  },
};
</script>
