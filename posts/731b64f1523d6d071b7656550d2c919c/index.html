		{
			printf("\n\n\n删除存储文件失败！！！\n");
			exit(0);
		}
		return; 
	}
	if ((fp = fopen("stu.dat", "wb+")) == NULL)
	{
		printf("\n\n\n文件打开失败！！！\n");
		return;
	}
	do
	{
		if ((fwrite(p0, sizeof(struct Student), 1, fp) != 1)) printf("\n\n\n文件写入失败！！！\n");
		else n++;
		p0 = p0->next;
		m++;
	} while (p0 != NULL );
	printf("\n\n\n应写入%d名学生信息，已写入%d名学生的信息！！！\n", m, n);
	fclose(fp);
}
