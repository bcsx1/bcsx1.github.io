1.2.2

实现基于二叉排序树的查找。根据提示输入参赛队编号，若查找成功，输出该赛事类别对应的基本信息（参赛作品名称、参赛学校、赛事类别、参赛者和指导老师信息），输出查找成功时的平均查找长度ASL，完成此操作，

首先建立一颗排序二叉树，需要将vector中的数据插入到排序二叉树中，设定一个函数函数接收两个参数，一个是指向二叉排序树根节点的指针 root，另一个是待插入的数据 data，如果当前二叉排序树为空，将新节点插入到二叉排序树的根节点位置，并设置左右孩子节点为 NULL。

如果当前节点不为空，则判断待插入节点的 id 是否比当前节点的 id 小或大，若小于则继续在左子树中寻找合适的位置进行插入，否则在右子树中查找。相当于使用递归的方式将待插入的节点依次比较并插入到对应的位置。这样就保证了二叉排序树中每个节点的 key 值大于其左子树所有节点的 key 值，小于其右子树所有节点的 key 值，从而使得二叉排序树具有良好的排序性质。

使用了指向指针的引用，即 BSTNode*&，这是为了确保在寻找插入位置时可以正确地修改树的结构。

在进行查找操作时，如果当前二叉排序树为空，则输出查找失败。如果当前节点不为空，则判断待查找的 id 是否比当前节点的 id 小或大，若小于则继续在左子树中查找，否则在右子树中查找。在查找过程中，需要累加查找次数并将累加字符串长度，以便后面计算 ASL。

如果找到了对应的节点，则输出该节点中数据的各个属性，并同样需要更新查找次数和累加字符串长度。

最终，通过查找次数和累加字符串长度计算出平均查找长度（ASL），即 访问路径的路径长度之和/对几个节点进行了访问。

    void search(const string& id) {

        sumLength = 0;
        countASL(m_root, id);
        double ASL = sumLength / a;
        if (m_root != NULL) {
            cout << "ASL(成功)=" << ASL << endl;
        }

    // 计算平均查找长度ASL
    int a = 0;
    double sumLength = 0;
    void countASL(BSTNode* root, const string& id) {
        if (root == NULL) { // 未找到节点
            cout << "查找失败！" << endl;
            return;
        }
        else if (id < root->data.id) { // 在左子树中查找
            a++;
            sumLength += id.length();
            countASL(root->left, id);
        }
        else if (id > root->data.id) { // 在右子树中查找
            a++;
            sumLength += id.length();
            countASL(root->right, id);
        }
        else { // 找到了节点
            cout << "作品名称：" << root->data.name << "\n参赛学校：" << root->data.school << "\n作品类别：" << root->data.category << "\n参赛者：" << root->data.player << "\n指导教师" << root->data.teacher << endl;
            a++;
            sumLength += id.length();
        }
    }

    // 插入节点到二叉排序树中
    void insertNode(BSTNode*& root, const Contestant& data) {
        if (root == NULL) { // 找到了插入位置
            root = new BSTNode;
            root->data = data;
            root->left = root->right = NULL;
        }
        else if (data.id < root->data.id) { // 在左子树中插入节点
            insertNode(root->left, data);
        }
        else if (data.id > root->data.id) { // 在右子树中插入节点
            insertNode(root->right, data);
        }
    }

赛事系统为参赛者提供赛地的校园导游程序。为参赛者提供各种路径导航的查询服务。完成该任务，需要将所有的地点和各个地点之间的距离设定好，使用了dijkstra算法，该算法以某个起始顶点为出发点，在加权图中找到从该顶点到其他所有顶点的最短路径，目的是计算出两个点之前最短的距离，并输出最短路径

struct Edge {
    int to, w;
    Edge(int to, int w) : to(to), w(w) {}
};

vector<Edge> G[MAXN];
int d[MAXN];
int p[MAXN];
void dijkstra(int s) {
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> q;
    memset(d, INF, sizeof(d));
    memset(p, -1, sizeof(p));
    d[s] = 0;
    q.push(make_pair(0, s));
    while (!q.empty()) {
        pair<int, int> x = q.top();
        q.pop();
        int u = x.second;
        if (d[u] < x.first) continue;
        for (int i = 0; i < G[u].size(); i++) {
            Edge e = G[u][i];
            if (d[e.to] > d[u] + e.w) {
                d[e.to] = d[u] + e.w;
                p[e.to] = u;
                q.push(make_pair(d[e.to], e.to));
            }
        }
    }
}

void print_path(int t) {
    if (p[t] == -1) {
        cout << t << "  ";
        return;
    }
    print_path(p[t]);
    cout << t << "  ";
}

    //插入排序
    void insort()
    {
        //插入排序
        //获取待排序学生数量，并使用外层循环遍历容器
        const size_t len = matched_students.size();
        //对于每个外层循环对应的 i 值，内层循环从外层循环的下一个元素开始依次遍历到尾部元素，找到最小值对应的索引 min_idx
        for (size_t i = 0; i < len - 1; ++i) {
            size_t min = i;
            for (size_t j = i + 1; j < len; ++j) {
                if (matched_students[j].getId() < matched_students[min].getId()) {
                    min = j;
                }
            }
            //如果遍历完整个内层循环后发现最小值不在起始位置处，就使用 std::swap 函数交换起始位置和找到的最小值位置上的学生信息
            if (min != i) {
                std::swap(matched_students[i], matched_students[min]);
            }
        }
    }

    void countASL(BSTNode* root, const string& id) {
        if (root == NULL)
        { // 未找到节点
            cout << "查找失败！" << endl;
            return;
        }
        else if (id < root->data.id)
        { // 在左子树中查找
            a++;
            sumLength += id.length();
            countASL(root->left, id);
        }
        else if (id > root->data.id) 
        { // 在右子树中查找
            a++;
            sumLength += id.length();
            countASL(root->right, id);
        }
        else 
        { // 找到了节点
            cout << "作品名称：" << root->data.name << "\n参赛学校：" << root->data.school << "\n作品类别：" << root->data.category << "\n参赛者：" << root->data.player << "\n指导教师" << root->data.teacher << endl;
            a++;
            sumLength += id.length();
        }
    }

    // 插入节点到二叉排序树中
    void insertNode(BSTNode*& root, const Contestant& data)
    {
        if (root == NULL)
        { // 找到了插入位置
            root = new BSTNode;
            root->data = data;
            root->left = root->right = NULL;
        }
        else if (data.id < root->data.id)
        { // 在左子树中插入节点
            insertNode(root->left, data);
        }
        else if (data.id > root->data.id)
        { // 在右子树中插入节点
            insertNode(root->right, data);
        }
    }

const int MAXN = 1005;
const int INF = 10000;

//记录顶点编号和边权重
struct Edge
{
    int to, w;
    Edge(int to, int w) : to(to), w(w) {}
};

vector<Edge> G[MAXN];
int d[MAXN];
int p[MAXN];
void dijkstra(int s) 
{
    //pair<int, int>表示元素类型，即每个顶点及其对应的起点到该顶点的距离
    //vector<pair<int, int>>表示底层容器类型，即使用vector来存储元素
    //greater<pair<int, int>>表示比较函数对象，定义元素之间的大小关系
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> q;
    //将整个d数组赋值为INF，各个顶点到起点的距离都初始设为无穷大
    memset(d, INF, sizeof(d));
    //memset(p, -1, sizeof(p)); 作用是将整个p数组赋值为-1
    memset(p, -1, sizeof(p));
    d[s] = 0;
    q.push(make_pair(0, s));
    //判断优先队列是否为空
    while (!q.empty())
    {
        pair<int, int> x = q.top();
        q.pop();
        int u = x.second;
        if (d[u] < x.first) continue;
        for (int i = 0; i < G[u].size(); i++)
        {
            Edge e = G[u][i];
            if (d[e.to] > d[u] + e.w)
            {
                d[e.to] = d[u] + e.w;
                p[e.to] = u;
                q.push(make_pair(d[e.to], e.to));
            }
        }
    }
}

void print_path(int t)
{
    //当t是根节点时，即t的父亲为-1时，输出t
    if (p[t] == -1) 
    {
        cout << t << "  ";
        return;
    }
    //最后输出节点t，输出从根节点到t这一段路径
    print_path(p[t]);
    cout << t << "  ";
}

            cout << "请合法输入!" << endl;
            return 0;
            break;
        }
        system("pause");
        system("cls");
    }

    return 0;
}
