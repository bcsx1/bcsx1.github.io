			$('body').append("<input type='text'/>")

			$("input").keydown(function(event) {

				if (event.keyCode == 13) {
					add();
				}

			});

			function add() {
				var nr="<li>"+$('input').val()+"<input type='button' name='' id='' value='↑' οnclick='up()'/><input type='button' name='' id='' value='↓' οnclick='down()'/><input type='button' name='' id='' value='删除' οnclick='del()'/></li>"
				$("input[type='text']").after(nr)
			}
			function up(){
				
				$f=$(event.target).parent();
				$f.prev().before($f)
				
			}
			function down(){
				
				$f=$(event.target).parent();
				$f.next().after($f)
				
			}
			function del(){
				$f=$(event.target).parent();
				$f.remove();
			}
		</script>
	</body>
</html>
