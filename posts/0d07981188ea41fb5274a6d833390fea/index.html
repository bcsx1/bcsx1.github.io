        逆向读取字符串： 

def reverseWords(input):

    # 通过空格将字符串分隔符，把各个单词分隔为列表
    inputWords = input.split(" ")

    # 翻转字符串
    # 假设列表 list = [1,2,3,4],
    # list[0]=1, list[1]=2 ，而 -1 表示最后一个元素 list[-1]=4 ( 与 list[3]=4 一样)
    # inputWords[-1::-1] 有三个参数
    # 第一个参数 -1 表示最后一个元素
    # 第二个参数为空，表示移动到列表末尾
    # 第三个参数为步长，-1 表示逆向
    inputWords = inputWords[-1::-1]
    print(inputWords)
    # 重新组合字符串
    output = ' '.join(inputWords)

    return output

if __name__ == "__main__":
    input = 'f11 q22 l33'
    rw = reverseWords(input)
    print(rw)

Tuple（元组）

Set（集合） 

        创建集合：

集合名 = {值1,值2,...}
或者
set(值) 


sites = {'1', '2', 'fql', '3', '4', '5'}

print(sites)  # 输出集合，重复的元素被自动去掉

# 成员测试
if 'fql' in sites :
    print('fql 在集合中')
else :
    print('fql 不在集合中')


# set可以进行集合运算
a = set('abracadabra')
b = set('alacazam')

print(a)

print(a - b)  # a 和 b 的差集

print(a | b)  # a 和 b 的并集

print(a & b)  # a 和 b 的交集

print(a ^ b)  # a 和 b 中不同时存在的元素
