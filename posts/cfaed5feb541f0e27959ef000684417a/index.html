（4）构建第一个vue应用

一、功能介绍

todoList：任务计划列表
大致功能：
1.输入框输入内容后，按enter键，即可把内容添加到下方列表
2.可将单个todo标为完成
3.可删除单个todo
4.双击todo进行编辑，按esc键取消
5.显示未完成的todo数量
6.可筛选未完成的todo，已完成的todo
7.可全部标为完成，可一次性清楚全部已完成todo，亦可一次性清楚全部todo

4、JS代码

<script>
    let id = 0;
    var app = new Vue({
        el: '#todo-app',
        data: function () {
            return {
              todos: [],
              newTodoTitle: '',
              editedTodo: null,
              intention: 'all',
            }
        },
        methods: {
          // 添加
          addTodo: function() {
            this.todos.push({id: id++, title: this.newTodoTitle, completed: false});
            this.newTodoTitle = '';
          },
          // 标记完成
          asCompleted: function(todo) {
            todo.completed = true
          },
          // 删除
          removeTodo: function(todo) {
            this.todos.splice(this.todos.indexOf(todo), 1)
          },
          // 编辑
          editTodo: function (todo) {
              this.editedTodo = {id: todo.id, title: todo.title}
          },
          // 编辑完成
          editDone: function(todo) {
            this.editedTodo = null
          },
          // 取消编辑
          cancelEdit: function(todo) {
            todo.title = this.editedTodo.title;
            this.editedTodo = null
          },
          // 全部完成
          markAllCompleted: function() {
            this.todos.map(function (todo) {
              if (!todo.completed) {
                todo.completed = true
              }
            })
          },
          // 清除已完成
          clearCompleted: function() {
            this.todos = this.todos.filter(todo => !todo.completed)
          },
          // 清除全部
          clearAll: function() {
            this.todos = []
          }
        },
        directives: {
          focus: {
            inserted: function (el) {
              el.focus()
            }
          }
        },
        computed: {
          leftTodos: function() {
            return this.todos.filter(todo => !todo.completed)
          },
          // 未完成的数量
          leftTodosCount: function() {
            return this.leftTodos.length
          },
          // 过滤已完成和进行中
          filteredTodos: function() {
            if (this.intention === 'ongoing') {
              return this.leftTodos
            } else if (this.intention === 'completed') {
              return this.todos.filter(todo => todo.completed)
            } else {
              return this.todos
            }
          }
        }
    })
</script>
