测试代码（TestC3P0）：

package com.yuanye.lesson05;

import com.yuanye.lesson05.utils.JdbcUtils_C3P0;


import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class TestC3P0 {
    public static void main(String[] args) {
        Connection conn = null;
        PreparedStatement st = null;

        try {
            conn = JdbcUtils_C3P0.getConnection();//原来是自己实现的，现在用别人实现的

            //区别
            //使用？占位符代替参数
            String sql = "insert into users(id,`name`,`password`,`email`,`birthday`) values(?,?,?,?,?)";

            st = conn.prepareStatement(sql);//()需要参数，预编译SQL,先写sql，然后不执行

            //手动给参数赋值
            st.setInt(1,6);//给id赋值
            st.setString(2,"yuanye4");
            st.setString(3,"123456");
            st.setString(4,"123456@qq.com");
            //注意点：sql.Date  数据库用的
            //      util.Date  JAVA用的  new Date().getTime()获得时间戳
            st.setDate(5,new java.sql.Date(new java.util.Date().getTime()));

            //执行
            int i = st.executeUpdate();
            if (i>0){
                System.out.println("插入成功！");
            }
        } catch (SQLException throwables) {
            throwables.printStackTrace();
        }finally {
            JdbcUtils_C3P0.release(conn,st,null);
        }
    }
}
