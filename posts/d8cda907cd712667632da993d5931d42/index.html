DarkNet源码解析——Makefile

本文在解析DarkNet的同时通过注释的方式对其中用到的Makefile相关语法进行讲解。

GPU=0 #定义GPU
CUDNN=0#定义是否使用CUDNN加速,cuDNN是用于深度神经网络的GPU加速库
OPENCV=1#是否使用OPENCV库,
OPENMP=1#用于共享内存并行系统的多处理器程序设计的一套编译处理反感
DEBUG=1#是否能用于调试,即是否gcc -g

ARCH= -gencode arch=compute_30,code=sm_30 \
      -gencode arch=compute_35,code=sm_35 \
      -gencode arch=compute_50,code=[sm_50,compute_50] \
      -gencode arch=compute_52,code=[sm_52,compute_52]
 #使用nvcc编译时使用 
#      -gencode arch=compute_20,code=[sm_20,sm_21] \ This one is deprecated?

# This is what I use, uncomment if you know your arch and want to specify
# ARCH= -gencode arch=compute_52,code=compute_52

VPATH=./src/:./examples#定义源码路径,make指令会先从当前目录中寻找,如果找不到会在所定义的源码路径中寻找
SLIB=libdarknet.so#定义动态库
ALIB=libdarknet.a#定义静态库
EXEC=darknet#定义可执行文件
OBJDIR=./obj/#定义目标文件目录

CC=gcc#定义C语言编译器,即gcc
CPP=g++#定义c++编译器
NVCC=nvcc #定义nvdia编译器
AR=ar#创建静态库(.a)文件指令
ARFLAGS=rcs#创建静态库所用配置文件
#参数r：在库中插入模块（替换）。当插入的模块名已经在库中存在，则替换同名的模块。如果若干模块中有一个模块在库中不存在，ar显示一个错误消息，并不替换其他同名模块。默认的情况下，新的成员增加在库的结尾处，可以使用其他任选项来改变增加的位置 
#参数c：创建一个库。不管库是否存在，都将创建。
#参数s：创建目标文件索引，这在创建较大的库时能加快时间。（补充：如果不需要创建索引，可改成大写S参数；如果。a文件缺少索引，可以使用ranlib命令添加）
OPTS=-Ofast#配置gcc使用的参数
LDFLAGS= -lm -pthread #链接时使用的库文件 libm.so用于数学计算,libpthread.so用于多线程
COMMON= -Iinclude/ -Isrc/ #指定头文件路径
CFLAGS=-Wall -Wno-unused-result -Wno-unknown-pragmas -Wfatal-errors -fPIC
#Wall	生成所有警告信息。
#-Wno 关闭unused警告.如unknown-pragmas：使用了不允许的#pragma
#-WFATAL-ERRORS 出现错误停止编译
#-fPIC 生成位置无关的代码。


ifeq ($(OPENMP), 1) #如果定义OPENMP,新增编译选项,即是否使用多线程
#ifeq为Makefile判断条件使用的语法
CFLAGS+= -fopenmp
endif

ifeq ($(DEBUG), 1) #如果需要调试程序,需要加上-g选项,gcc在源码中加入调试信息.否则不能定位到程序运行位置
OPTS=-O0 -g
endif

CFLAGS+=$(OPTS)#添加编译时使用的参数

ifeq ($(OPENCV), 1) #调用opencv相关的库文件以及头文件
COMMON+= -DOPENCV#-D选项是用来在使用gcc/g++编译的时候定义宏的。
CFLAGS+= -DOPENCV
LDFLAGS+= `pkg-config --libs opencv` -lstdc++#查找opencv依赖包(--libs 查找库文件)
COMMON+= `pkg-config --cflags opencv` #pkg-config 当你从源代码编译软件时，用来提供依赖库的信息的软件(查找头文件)
endif

ifeq ($(GPU), 1) #GPU相关配置
COMMON+= -DGPU -I/usr/local/cuda/include/
CFLAGS+= -DGPU
LDFLAGS+= -L/usr/local/cuda/lib64 -lcuda -lcudart -lcublas -lcurand
endif

ifeq ($(CUDNN), 1) #配置cudnn库以及宏定义CUDNN
COMMON+= -DCUDNN 
CFLAGS+= -DCUDNN
LDFLAGS+= -lcudnn
endif
#生成目标.o文件列表
OBJ=gemm.o utils.o cuda.o deconvolutional_layer.o convolutional_layer.o list.o image.o activations.o im2col.o col2im.o blas.o crop_layer.o dropout_layer.o maxpool_layer.o softmax_layer.o data.o matrix.o network.o connected_layer.o cost_layer.o parser.o option_list.o detection_layer.o route_layer.o upsample_layer.o box.o normalization_layer.o avgpool_layer.o layer.o local_layer.o shortcut_layer.o logistic_layer.o activation_layer.o rnn_layer.o gru_layer.o crnn_layer.o demo.o batchnorm_layer.o region_layer.o reorg_layer.o tree.o  lstm_layer.o l2norm_layer.o yolo_layer.o iseg_layer.o image_opencv.o
#生成目标.o文件列表
EXECOBJA=captcha.o lsd.o super.o art.o tag.o cifar.o go.o rnn.o segmenter.o regressor.o classifier.o coco.o yolo.o detector.o nightmare.o instance-segmenter.o darknet.o
ifeq ($(GPU), 1) 
LDFLAGS+= -lstdc++ 
OBJ+=convolutional_kernels.o deconvolutional_kernels.o activation_kernels.o im2col_kernels.o col2im_kernels.o blas_kernels.o crop_layer_kernels.o dropout_layer_kernels.o maxpool_layer_kernels.o avgpool_layer_kernels.o
endif
#addprefix添加前缀,用于查找EXECOBJA文件列表
EXECOBJ = $(addprefix $(OBJDIR), $(EXECOBJA))
OBJS = $(addprefix $(OBJDIR), $(OBJ))
#wildcard用于展开文件夹中符合pattern的文件
DEPS = $(wildcard src/*.h) Makefile include/darknet.h
#目标文件:依赖文件
#make时默认执行该语句,即创建obj,backup,results文件夹以及编译$(SLIB) $(ALIB) $(EXEC)
all: obj backup results $(SLIB) $(ALIB) $(EXEC)
#all: obj  results $(SLIB) $(ALIB) $(EXEC)
#根据上述配置选项进行编译.
#$@  表示目标文件
#$^  表示所有的依赖文件

#生成执行文件
$(EXEC): $(EXECOBJ) $(ALIB)
	$(CC) $(COMMON) $(CFLAGS) $^ -o $@ $(LDFLAGS) $(ALIB)
#生成静态库
$(ALIB): $(OBJS)
	$(AR) $(ARFLAGS) $@ $^
#生成动态库
$(SLIB): $(OBJS)
	$(CC) $(CFLAGS) -shared $^ -o $@ $(LDFLAGS)

$(OBJDIR)%.o: %.cpp $(DEPS)
	$(CPP) $(COMMON) $(CFLAGS) -c $< -o $@

$(OBJDIR)%.o: %.c $(DEPS)
	$(CC) $(COMMON) $(CFLAGS) -c $< -o $@

$(OBJDIR)%.o: %.cu $(DEPS)
	$(NVCC) $(ARCH) $(COMMON) --compiler-options "$(CFLAGS)" -c $< -o $@

obj:
	mkdir -p obj
backup:
	mkdir -p backup
results:
	mkdir -p results

.PHONY: clean

clean:
	rm -rf $(OBJS) $(SLIB) $(ALIB) $(EXEC) $(EXECOBJ) $(OBJDIR)/*
