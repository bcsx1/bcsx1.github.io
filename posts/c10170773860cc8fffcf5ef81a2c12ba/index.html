void HPpush(HP* hp, HPDatatype x)
{
	assert(hp);
	if (hp->HPcapacity == hp->size)//如果数组里面什么都没有，没有栈顶元素也没有容量，也就是top和capacity都=0，,那么就扩容
	{
		int newcapcity = hp->HPcapacity == 0 ? 4 : hp->HPcapacity * 2;

	assert(hp);
	return  hp->size == 0;

void AdjustDown(int* a, int n, int parent)
{
	int child = parent * 2 + 1;
	while (child < n)
	{
		// 选出左右孩子中小/大的那个
		if (child + 1 < n
			&& a[child + 1] > a[child])
		{
			++child;
		}

		if (a[child] > a[parent])
		{
			swap(&a[parent], &a[child]);
			parent = child;
			child = parent * 2 + 1;
		}
		else
		{
			break;
		}
	}
}

void HPpop(HP* hp)
{
	assert(hp);
	swap(&hp->a[0], &hp->a[hp->size-1]);

	hp->size--;	
	AdjustDown(hp->a, hp->size, 0);
}

	for (int i = 0; i < sizeof(a) / sizeof(int); ++i)
	{


		HPpush(&tr, a[i]);
	}

	HPpop(&tr);
	return 0;
}

void HPqsort(int* a, int n)
{//向上调整

	for (int i = 1; i < n; i++)
	{
ADjustup(a, i);
	}
	

	//向下调整
	int end = n - 1;
	while (end)
	{
		swap(&a[0],&a[end]);
		AdjustDown(a,  end,  0);
		--end;
	}
}
