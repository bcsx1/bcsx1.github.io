export default class MethodModal extends PureComponent {

  constructor(props) {
    super(props);
  }

  state = {
    tempIndex:0,
  };

  componentDidMount() {
    const {dispatch } = this.props;
  }

  componentWillReceiveProps(props) {
    
  }


  handleChangeSelect = val => {

    const {
      form: { getFieldValue },
      dispatch,
    } = this.props;

    const argVal = getFieldValue('value');

    var index = val.index;

    argVal[index].name = val.name;

    delete argVal[index].key;
    delete argVal[index].expression;
    delete argVal[index].expect;
    
    this.props.onChange(argVal);

  };

  add = () => {
    const {tempIndex}= this.state;
    const argVal = this.props.form.getFieldValue('value');
    argVal.push({name:''});

    this.setState({tempIndex: tempIndex + 1})
    this.props.onChange(argVal);

  };



  addPath = index => {
    const {
      form: { getFieldValue },
      dispatch,
    } = this.props;

    const argVal = getFieldValue('value');
    argVal[index]['jsonPath'] = 'result[*]';

    this.props.onChange(argVal);
    
  }


  remove = index => {
    const { form } = this.props;
    const argVal  = form.getFieldValue('value');
    argVal.splice(index, 1);
    form.resetFields(['value'])
    this.props.onChange(argVal);
  };


  render () {

    const {
      form: { getFieldDecorator },
      dispatch,
    } = this.props;


    const renderHidden = (name, value,index) => {
      let hiddenKey = index + "_" + name;
      return (
        <Col key={hiddenKey} span={1}>
          <FormItem>
            {getFieldDecorator(name, {
              initialValue: value,
            })(<Input style={{ width: '80%' }} hidden />)}
          </FormItem>
        </Col>
      );
    };

    const renderHiddenPath = (name, method,span,index) => {
      let hiddenPathKey = index + "_" + name;

        return (
          <Tooltip title="校验路径jsonPath">
              {method.jsonPath && (
                <Col key={hiddenPathKey+"_path"} span={span}>
                  <FormItem>
                    {getFieldDecorator(name, {
                      initialValue: method.jsonPath,
                    })(<Input style={{ width: '80%' }} />)}
                  </FormItem>
                </Col>
              )}
          {!method.jsonPath && (
            <Col span={1} key={hiddenPathKey+"_npath"} style={{ margin: '6px auto' }}>
              <Icon
                type="ellipsis"
                style={{ fontSize: '18px' }}
                onClick={() => this.addPath(index)}
              />
          </Col>
         )}
        </Tooltip>
        );
    };

    const renderInput = (label, name, initValue,key) => {
      let itemKey = key + "_" + name
      return (
        <Col key={itemKey} span={4}>
          <FormItem>
            {getFieldDecorator(name, {
              initialValue: initValue,
              normalize: value => {
                return value;
              },
            })(<Input style={{ width: '80%' }} placeholder={label} />)}
          </FormItem>
        </Col>
      );
    };

  const renderSelect = (label, name, opts, initValue,key) => {
    
      let selectKey = key + "_" + name

      return (
        <Col key={selectKey} span={4}>
          <FormItem>
            {getFieldDecorator(name, {
              initialValue: initValue
            })(
              <Select style={{ width: '80%' }}>
                {opts.map((val, ind) => {
                  return (
                    <Select.Option value={val} key={ind}>
                      {val}
                    </Select.Option>
                  );
                })}
              </Select>
            )}
          </FormItem>
        </Col>
      );
    };

    const renderSelectMethod = (methodItem, index) => {
      let colKey = index + '_method';

      return (
        <Col key={colKey} span={5}>
          <FormItem>
            {getFieldDecorator(`value[${index}][name]`, {
              initialValue:methodItem.name,
            })(
              <Select
                style={{ width: '80%' }}
                onChange={value => {
                  this.handleChangeSelect({ name:value, index: index});
                }}
              >
                   <Select.Option value='' key=''>
                      请选择校验规则
                   </Select.Option>

                  {this.props.methodList.map((method, index) => {
                    return (
                      <Select.Option value={method.name} key={method.name}>
                        {method.nickName}
                      </Select.Option>
                    );
                  })}

              </Select>
              
            )}
          </FormItem>
        </Col>
      );
    };

    const renderItems = (method, index0) => {

      if (!method || !method.name) {
        return <span></span>;
      }

      const runnerMethodArgs = this.props.methodListMap.get(method.name).runnerMethodArgs;

      if (!runnerMethodArgs) {
        return <span></span>;
      }


      return (
        <div>
          {runnerMethodArgs.map((argDO, index) => {
            
            let initValue = method[argDO.name]
            
            if (argDO.optional === '[]') {
              return renderInput(
                argDO.nickName,
                `value[${index0}][${argDO.name}]`,
                initValue,
                index0
              );
            } else if (argDO.optional !== '[]') {
              try {
                if (!(argDO.optional instanceof Array)) {
                  argDO.optional = JSON.parse(argDO.optional);
                }
              } catch (e) {
                argDO.optional = [];
              }

              return renderSelect(
                argDO.nickName,
                `value[${index0}][${argDO.name}]`,
                argDO.optional,
                initValue,
                index0
              );
            }
          })}
        </div>
      );
    };

    let value = this.props.value;
    if (value.length < 1) {
      value.push({ name:'',nickName: '请选择校验规则'});
    }
    
    return (
      <div>
        {value.map((method, index0) => {
          return (
            <Row key={index0}>
              
              {renderSelectMethod(method, index0)}

              {renderItems(method, index0)}
            
              {renderHiddenPath(`value[${index0}][jsonPath]`, method,3,index0)}

              <Col style={{ margin: '5px auto' }} span={2}>
       
                <Icon
                  type="minus-circle-o"
                  style={{ fontSize: '18px' }}
                  onClick={() => this.remove(index0)}
                />

                <Icon
                  type="plus-circle-o"
                  style={{ fontSize: '18px' }}
                  onClick={() => this.add()}
                />
               
              </Col>

            </Row>
          );
        })}
      </div>
    );
  }
}
