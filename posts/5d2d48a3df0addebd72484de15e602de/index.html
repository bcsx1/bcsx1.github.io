  # math_get_square_root
  # Input: int
  # Output: reg1 
  ("math_get_square_root", [
     (store_script_param, ":val", 1),
     (assign, ":flag", 0),
     (try_for_range, ":index", 0, 10),
        (store_mul, ":square_rst", ":index", ":index"),
        (try_begin),
           (gt, ":square_rst", ":val"),
           (eq, ":flag", 0),
           (store_sub, ":sqare_root_rst", ":index", 1),
           (assign, reg1, ":sqare_root_rst"),
           (assign, ":flag", 1),
        (try_end),
     (try_end),
  ]),
