如果想要存储下面的自定义的数据类型，PlayerPrefs非常难以实现，并且PlayerInfo中含有另一种自定义类型（ItemInfo)。

public class PlayerInfo
{
    public int age = 10;
    public string name = "未命名";
    public float height = 177.5f;
    public bool sex = true;
    public List<int> list = new List<int>() { 1, 2, 3, 4 };
    public Dictionary<string, int> dic = new Dictionary<string, int>() { { "11", 1 }, { "22", 2 } };
    public ItemInfo itemInfo = new ItemInfo();
    public List<ItemInfo> itemList = new List<ItemInfo>() { new ItemInfo(1, 10), new ItemInfo(2, 20) };
    public Dictionary<int, ItemInfo> itemDic = new Dictionary<int, ItemInfo>() { { 3, new ItemInfo(3, 30) }, { 4, new ItemInfo(4, 40) } };
}

public class ItemInfo
{
    public int id;
    public int num;
    public ItemInfo()
    {
    }
    public ItemInfo(int id, int num)
    {
        this.id = id;
        this.num = num;
    }
}

我利用C#的反射机制以及递归思想，实现了可以存储自定类型（里面可以嵌套其它的自定义类型，List,Dictionary<Key,Value>等常用容器）的PlayerPrefs管理类。



using System;
using System.Collections;
using System.Reflection;
using UnityEngine;

//PlayerPrefs管理类 统一管理数据的存储和读取
public class PlayerPrefsMrg 
{
    //单例模式 
    private static PlayerPrefsMrg instance = new PlayerPrefsMrg();
    public static PlayerPrefsMrg Instance => instance;
    private PlayerPrefsMrg()
    {
    }

    //存储对象  data为传入对象 keyName为存储名字也就是数据对象的唯一key
    public void SaveData(object data,string keyName)
    {
        Type dataType = data.GetType();
        //得到所有公共字段
        FieldInfo[] fieldInfos = dataType.GetFields();
        //定义一个key规则 来保证key的唯一性
        //比如 keyname_数据类型_字段类型_字段名

        //遍历这些字段 进行数据存储
        string savaKeyName = "";
        FieldInfo info;
        for (int i=0;i<fieldInfos.Length;i++)
        {
            //对每一个字段进行存储
            info = fieldInfos[i];
            //可以通过FieldInfo可以直接获取到 公共字段类型和字段的名字
            //字段的类型名 info.FieldType.Name
            //字段的名字 info.Name
            //Player1_Player_Int32_age
            //定义的key规则 keyname_数据类型_字段类型_字段名
            savaKeyName = keyName + "_" + dataType.Name + "_" + info.FieldType.Name + "_" + info.Name;

            //现在得到key 按照我们的规则
            //接下来就是要通过PlayerPrefs来进行存储
            //获取值  info.GetValue(data)方法（获取Player类型指定字段的某个Player对象的值）
            //info只代表字段类型
            SaveValue(info.GetValue(data), savaKeyName);

        }


       
    }

    private void SaveValue(object value,string saveKeyName)
    {
        //使用PlayerPrefs进行存储
        Type fieldType = value.GetType();
        if (fieldType == typeof(int))
        {
            
            PlayerPrefs.SetInt(saveKeyName, (int)value);
            //为int数据加密
            //PlayerPrefs.SetInt(saveKeyName, (int)value + 10);
        }
        else if (fieldType == typeof(float))
        {
            PlayerPrefs.SetFloat(saveKeyName, (float)value);
        }
        else if (fieldType == typeof(string))
        {
            PlayerPrefs.SetString(saveKeyName, value.ToString());
        }
        else if(fieldType==typeof(bool))
        {
            //自己定义一个存储bool的规则
            PlayerPrefs.SetInt(saveKeyName, (bool)value ? 1 : 0);
        }
        //判断泛型(list)： 通过判断父子关系来实现。相当于是判断 字段是不是IList的子类
        else if(typeof(IList).IsAssignableFrom(fieldType))
        {
            //父类装子类
            IList list = value as IList;
            PlayerPrefs.SetInt(saveKeyName, list.Count);
            int index = 0;//计数器
            foreach(object item in list)
            {
                //递归反复判断
                SaveValue(item, saveKeyName + index++);
            }
        }
        //判断泛型(Dictionary):
        else if(typeof(IDictionary).IsAssignableFrom(fieldType))
        {
            //父类装子类
            IDictionary dic = value as IDictionary;
            //先存储长度
            PlayerPrefs.SetInt(saveKeyName, dic.Count);
            int index = 0;//用于计数
            foreach(object  key in dic.Keys)
            {
                //区分key和value(主要是预防DIctionary中的两个泛型类型相同）
                SaveValue(key, saveKeyName + "_key_" + index);
                SaveValue(dic[key], saveKeyName + "_value_" + index);
                index++;
            }
        }
        //自定义类型 通过递归实现嵌套的数据存储
        else
        {
            SaveData(value, saveKeyName);
        }
    }

    //读取数据  type为传入类型，name为存储名字也就是数据对象的唯一key， 直接返回一个对象
    public object LoadData(Type type,string keyName)
    {
        //不用object对象传入 而使用Type传入
        //主要目的是在外部节约一行代码
        object data = Activator.CreateInstance(type);
        FieldInfo[] fieldInfos = type.GetFields();
        FieldInfo info;
        string readKeyName = " ";
        for(int i=0;i<fieldInfos.Length;i++)
        {
            info = fieldInfos[i];
            readKeyName = keyName + "_" + type.Name + "_" + info.FieldType.Name + "_" + info.Name;
            //填充数据到data对象中
            info.SetValue(data, ReadValue(info.FieldType, readKeyName));
        }
        
        return data;
    }

    public object ReadValue(Type fieldType,string keyName)
    {
        
        if (fieldType == typeof(int))
        {
            
            return PlayerPrefs.GetInt(keyName);
            //为Int数据解密
            //return PlayerPrefs.GetInt(keyName) - 10;
        }
        else if (fieldType == typeof(float))
        {
            
            return PlayerPrefs.GetFloat(keyName);
        }
        else if (fieldType == typeof(string))
        {
            return PlayerPrefs.GetString(keyName);
        }
        else if (fieldType == typeof(bool))
        {
            return PlayerPrefs.GetInt(keyName) == 0 ? false : true;
        }
        else if (typeof(IList).IsAssignableFrom(fieldType))
        {
            int count = PlayerPrefs.GetInt(keyName);
            //实例化一个List对象 来进行赋值
            IList list = Activator.CreateInstance(fieldType) as IList;
            for(int i=0;i<count;i++)
            {
                //得到list泛型的类型,通过GetGenericArguments()获取泛型
                list.Add(ReadValue(fieldType.GetGenericArguments()[0], keyName + i));
            }
            return list;
        }
        else if(typeof(IDictionary).IsAssignableFrom(fieldType))
        {
            //获取数量
            int count = PlayerPrefs.GetInt(keyName);
            //实例化一个字典
            IDictionary dictionary = Activator.CreateInstance(fieldType) as IDictionary;
            for(int i=0;i<count;i++)
            {
                //获取字典中对应的键值对
                dictionary.Add(ReadValue(fieldType.GetGenericArguments()[0], keyName + "_key_" + i),
                               ReadValue(fieldType.GetGenericArguments()[1], keyName + "_value_" + i));
                
            }
            return dictionary;
        }
        else
        {
            return LoadData(fieldType, keyName);
        }
    }
}

using UnityEngine;
public class Test : MonoBehaviour
{

    void Start()
    {
        //实例化一个对象
        PlayerInfo player1 = new PlayerInfo();
        
        //利用PlaryerPrefsMrg.SaveData()方法进行数据存储
        PlayerPrefsMrg.Instance.SaveData(player1, "player1");
        
		//利用PlaryerPrefsMrg.ReadData()方法进行数据读取
        PlayerInfo test = PlayerPrefsMrg.Instance.LoadData(typeof(PlayerInfo), "player1") as PlayerInfo;
        print(test.age);
        print(test.name);
        foreach (var item in test.list)
        {
            print(item);
        }
        foreach(ItemInfo item in test.itemList)
        {
            print(item.id);
            print(item.num);
        }
    }
}
