   // 实现样式切换
  <script> 
      //获取所有的 li标签
      let list = document.querySelectorAll(".navigation li");
      //创建样式切换事件
      function activeLink() {
         // 遍历所有的li标签
        list.forEach((item) => {
         // 删除所有的active
          item.classList.remove("active");
        });
        //  只对选中的添加active
        this.classList.add("active");
      }
      // 为所有的li 标签添加点击事件
      list.forEach((item) => {
        item.addEventListener("click", activeLink);
      });
      console.log("list", list);
  </script>
